47  std::vector<std::string> get_default_languages()
48  {
49  // First try parsing the LANGUAGE environment variable.
50  // This is the best, because it lets the user specify a list of languages
51  // in descending priority, so if a translation is unavailable in their
52  // primary language, they may still get another match which would be
53  // better for them than the default enUS (e.g., a French person could
54  // prefer to fall back on a Spanish translation rather than English).
55  const std::string sLanguageVar = get_environment_variable("LANGUAGE");
56  if (!sLanguageVar.empty())
57  {
58  std::vector<std::string> lOutput;
59  for (auto sLanguage : utils::cut(sLanguageVar, ":"))
60  {
61  std::string sLanguageNormalized{sLanguage};
62  utils::replace(sLanguageNormalized, "_", "");
63  if (sLanguageNormalized.size() == 4)
64  lOutput.push_back(sLanguageNormalized);
65  }
66 
67  if (!lOutput.empty())
68  return lOutput;
69  }
70 
71  #if defined(LXGUI_PLATFORM_WINDOWS)
72  // If LANGUAGE is not specified, on Windows, try OS-specific function.
73  // TODO: https://github.com/cschreib/lxgui/issues/95
74  #endif
75 
76  // If LANGUAGE is not specified or empty, try LANG.
77  std::string sLang = get_environment_variable("LANG");
78  if (!sLang.empty())
79  {
80  auto uiPos1 = sLang.find_first_of(".@");
81  if (uiPos1 != std::string::npos)
82  sLang = sLang.substr(0, uiPos1);
83 
84  utils::replace(sLang, "_", "");
85  if (sLang.size() == 4)
86  return {sLang};
87  }
88 
89  return {"enUS"};
90  }
91 }

94 {
95  try
96  {

104  gui::out << gui::error << "gui::locale : " << mException.what() << std::endl;
105  gui::out << gui::error << "gui::locale : reverting to default classic locale" << std::endl;
106  }
107 
108  // Find default languages.
109  auto_detect_preferred_languages();
110  auto_detect_allowed_code_points();
111 
112  // Set up Lua sandbox.
113  mLua_.open_libraries(
114  sol::lib::base, sol::lib::math, sol::lib::table, sol::lib::io,
115  sol::lib::os, sol::lib::string, sol::lib::debug
116  );
117 
118  // Give the translation Lua state the localize_string and format_string function, so
119  // it can call it recursively as needed, but not the other functions
120  // which could load more translation strings.
121  mLua_.set_function("localize_string", [&](const std::string& sKey, sol::variadic_args mVArgs)
122  {
123  return localize(sKey, mVArgs);
124  });
125  mLua_.set_function("format_string", [&](const std::string& sKey, sol::variadic_args mVArgs)
126  {
127  return format_string(sKey, mVArgs);
128  });

174 void localizer::add_allowed_code_points(const code_point_range& mRange)
175 {
176  if (mRange.uiLast < mRange.uiFirst)
177  throw gui::exception("gui::localizer", "code point range must have last >= first");
178 
179  code_point_range mTestRange = mRange;
180  auto mIter = lCodePoints_.begin();
181 
182  do
183  {
184  // Find next overlapping range
185  mIter = std::find_if(lCodePoints_.begin(), lCodePoints_.end(), [&](const auto& mOther)
186  {
187  return (mTestRange.uiFirst >= mOther.uiFirst && mTestRange.uiFirst <= mOther.uiLast) ||
188  (mTestRange.uiLast >= mOther.uiFirst && mTestRange.uiLast <= mOther.uiLast) ||
189  (mOther.uiFirst >= mTestRange.uiFirst && mOther.uiFirst <= mTestRange.uiLast) ||
190  (mOther.uiLast >= mTestRange.uiFirst && mOther.uiLast <= mTestRange.uiLast);
191  });
192 
193  if (mIter != lCodePoints_.end())
194  {
195  // Combine the ranges
196  mTestRange.uiFirst = std::min(mTestRange.uiFirst, mIter->uiFirst);
197  mTestRange.uiLast = std::max(mTestRange.uiLast, mIter->uiLast);
198 
199  // Erase the overlap
200  lCodePoints_.erase(mIter);
201  }
202  } while (mIter != lCodePoints_.end());
203 
204  // Add the new range
205  lCodePoints_.push_back(mTestRange);
206 
207  // Sort by ascending code point
208  std::sort(lCodePoints_.begin(), lCodePoints_.end(), [](const auto& mLeft, const auto& mRight)
209  {
210  return mLeft.uiFirst < mRight.uiFirst;
211  });
212 }

734 void localizer::load_translations(const std::string& sFolderPath)
735 {
736  // First, look for an exact match
737  for (const std::string& sLanguage : lLanguages_)
738  {
739  std::string sLanguageFile = sFolderPath + "/" + sLanguage + ".lua";
740  if (utils::file_exists(sLanguageFile))
741  {
742  load_translation_file(sLanguageFile);
743  return;
744  }
745  }
746 
747  // If no exact match found, look for an approximate match (ignore region)
748  const auto lFiles = utils::get_file_list(sFolderPath, false, "lua");
749  for (const std::string& sLanguage : lLanguages_)
750  {
751  auto mIter = std::find_if(lFiles.begin(), lFiles.end(), [&](const std::string& sFile)
752  {
753  return sFile.size() == 8u && sFile.substr(0, 2) == sLanguage.substr(0, 2);
754  });
755 
756  if (mIter == lFiles.end())
757  continue;
758 
759  std::string sLanguageFile = sFolderPath + "/" + *mIter;
760  load_translation_file(sLanguageFile);
761  return;
762  }
763 
764  // If no match found, fall back to US english
765  std::string sLanguageFile = sFolderPath + "/enUS.lua";
766  if (utils::file_exists(sLanguageFile))
767  {
768  load_translation_file(sLanguageFile);
769  return;
770  }
771 }

773 void localizer::load_translation_file(const std::string& sFilename) try
774 {
775  auto mResult = mLua_.safe_script_file(sFilename);
776  if (!mResult.valid())
777  {
778  sol::error mError = mResult;
779  gui::out << gui::error << "gui::locale : " << mError.what() << std::endl;
780  return;
781  }
782 
783  sol::table mTable = mLua_["localize"];
784  if (mTable == sol::lua_nil)
785  {
786  gui::out << gui::warning << "gui::locale : no 'localize' table in " << sFilename << std::endl;
787  return;
788  }
789 
790  mTable.for_each([&](const sol::object& mKey, const sol::object& mValue)
791  {
792  if (!mKey.is<std::string>()) return;
793  std::string ks = mKey.as<std::string>();
794 
795  if (mValue.is<std::string>())
796  lMap_.insert(std::make_pair(std::hash<std::string>{}(ks), mValue.as<std::string>()));
797  else if (mValue.is<sol::protected_function>())
798  lMap_.insert(std::make_pair(std::hash<std::string>{}(ks), mValue.as<sol::protected_function>()));
799  });
800 
801  // Keep a copy so variables/functions remain alive
802  mLua_["localize_" + std::to_string(std::hash<std::string>{}(sFilename))] = mTable;
803 }
804 catch (const sol::error& mError)
805 {
806  gui::out << gui::error << "gui::locale : " << mError.what() << std::endl;
807  return;
808 }

826 std::string localizer::format_string(std::string_view sMessage, sol::variadic_args mVArgs) const
827 {

829  for (auto&& mArg : mVArgs)
830  {
831  lxgui::utils::variant mVariant;
832  if (!mArg.is<sol::lua_nil_t>())
833  mVariant = mArg;
834 
835  std::visit([&](auto& mValue)
836  {
837  using inner_type = std::decay_t<decltype(mValue)>;
838  if constexpr (std::is_same_v<inner_type, lxgui::utils::empty>)
839  mStore.push_back(static_cast<const char*>(""));

842  }, mVariant);
843  }
844
