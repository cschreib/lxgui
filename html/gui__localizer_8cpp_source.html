55  std::vector<std::string> get_default_languages()
56  {
57  // First try parsing the LANGUAGE environment variable.
58  // This is the best, because it lets the user specify a list of languages
59  // in descending priority, so if a translation is unavailable in their
60  // primary language, they may still get another match which would be
61  // better for them than the default enUS (e.g., a French person could
62  // prefer to fall back on a Spanish translation rather than English).
63  const std::string sLanguageVar = get_environment_variable("LANGUAGE");
64  if (!sLanguageVar.empty())
65  {
66  std::vector<std::string> lLanguages = utils::cut(sLanguageVar, ":");
67  std::vector<std::string> lOutput;
68  for (auto& sLanguage : lLanguages)
69  {
70  utils::replace(sLanguage, "_", "");
71  if (sLanguage.size() == 4)
72  lOutput.push_back(sLanguage);
73  }
74 
75  if (!lOutput.empty())
76  return lOutput;
77  }
78 
79  #if defined(LXGUI_PLATFORM_WINDOWS)
80  // If LANGUAGE is not specified, on Windows, try OS-specific function.
81  // TODO
82  #endif
83 
84  // If LANGUAGE is not specified or empty, try LANG.
85  std::string sLang = get_environment_variable("LANG");
86  if (!sLang.empty())
87  {
88  auto uiPos1 = sLang.find_first_of(".@");
89  if (uiPos1)
90  sLang = sLang.substr(0, uiPos1);
91 
92  utils::replace(sLang, "_", "");
93  if (sLang.size() == 4)
94  return {sLang};
95  }
96 
97  return {"enUS"};
98  }

181 void localizer::add_allowed_code_points(const code_point_range& mRange)
182 {
183  if (mRange.uiLast < mRange.uiFirst)
184  throw gui::exception("gui::localizer", "code point range must have last >= first");
185 
186  code_point_range mTestRange = mRange;
187  auto mIter = lCodePoints_.begin();
188 
189  do
190  {
191  // Find next overlapping range
192  mIter = std::find_if(lCodePoints_.begin(), lCodePoints_.end(), [&](const auto& mOther)
193  {
194  return (mTestRange.uiFirst >= mOther.uiFirst && mTestRange.uiFirst <= mOther.uiLast) ||
195  (mTestRange.uiLast >= mOther.uiFirst && mTestRange.uiLast <= mOther.uiLast) ||
196  (mOther.uiFirst >= mTestRange.uiFirst && mOther.uiFirst <= mTestRange.uiLast) ||
197  (mOther.uiLast >= mTestRange.uiFirst && mOther.uiLast <= mTestRange.uiLast);
198  });
199 
200  if (mIter != lCodePoints_.end())
201  {
202  // Combine the ranges
203  mTestRange.uiFirst = std::min(mTestRange.uiFirst, mIter->uiFirst);
204  mTestRange.uiLast = std::max(mTestRange.uiLast, mIter->uiLast);
205 
206  // Erase the overlap
207  lCodePoints_.erase(mIter);
208  }
209  } while (mIter != lCodePoints_.end());
210 
211  // Add the new range
212  lCodePoints_.push_back(mTestRange);
213 
214  // Sort by ascending code point
215  std::sort(lCodePoints_.begin(), lCodePoints_.end(), [](const auto& mLeft, const auto& mRight)
216  {
217  return mLeft.uiFirst < mRight.uiFirst;
218  });
219 }

741 void localizer::load_translations(const std::string& sFolderPath)
742 {
743  // First, look for an exact match
744  for (const std::string& sLanguage : lLanguages_)
745  {
746  std::string sLanguageFile = sFolderPath + "/" + sLanguage + ".lua";
747  if (utils::file_exists(sLanguageFile))
748  {
749  load_translation_file(sLanguageFile);
750  return;
751  }
752  }
753 
754  // If no exact match found, look for an approximate match (ignore region)
755  const auto lFiles = utils::get_file_list(sFolderPath, false, "lua");
756  for (const std::string& sLanguage : lLanguages_)
757  {
758  auto mIter = std::find_if(lFiles.begin(), lFiles.end(), [&](const std::string& sFile)
759  {
760  return sFile.size() == 8u && sFile.substr(0, 2) == sLanguage.substr(0, 2);
761  });
762 
763  if (mIter == lFiles.end())
764  continue;
765 
766  std::string sLanguageFile = sFolderPath + "/" + *mIter;
767  load_translation_file(sLanguageFile);
768  return;
769  }
770 
771  // If no match found, fall back to US english
772  std::string sLanguageFile = sFolderPath + "/enUS.lua";
773  if (utils::file_exists(sLanguageFile))
774  {
775  load_translation_file(sLanguageFile);
776  return;
777  }
778 }

780 void localizer::load_translation_file(const std::string& sFilename) try
781 {
782  auto mResult = mLua_.safe_script_file(sFilename);
783  if (!mResult.valid())
784  {
785  sol::error mError = mResult;
786  gui::out << gui::error << "gui::locale : " << mError.what() << std::endl;
787  return;
788  }
789 
790  sol::table mTable = mLua_["localize"];
791  if (mTable == sol::lua_nil)
792  {
793  gui::out << gui::warning << "gui::locale : no 'localize' table in " << sFilename << std::endl;
794  return;
795  }
796 
797  mTable.for_each([&](const sol::object& mKey, const sol::object& mValue)
798  {
799  if (!mKey.is<std::string>()) return;
800  std::string ks = mKey.as<std::string>();
801 
802  if (mValue.is<std::string>())
803  lMap_.insert(std::make_pair(std::hash<std::string>{}(ks), mValue.as<std::string>()));
804  else if (mValue.is<sol::protected_function>())
805  lMap_.insert(std::make_pair(std::hash<std::string>{}(ks), mValue.as<sol::protected_function>()));
806  });
807 
808  // Keep a copy so variables/functions remain alive
809  mLua_["localize_" + std::to_string(std::hash<std::string>{}(sFilename))] = mTable;
810 }
811 catch (const sol::error& mError)
812 {
813  gui::out << gui::error << "gui::locale : " << mError.what() << std::endl;
814  return;
815 }

833 std::string localizer::format_string(std::string_view sMessage, sol::variadic_args mVArgs) const
834 {

840  mVariant = mArg;

845  if constexpr (std::is_same_v<inner_type, lxgui::utils::empty>)
846  mStore.push_back(static_cast<const char*>(""));
