298 void status_bar::update(float fDelta)
299 {
300 
301  if (bUpdateBarTexture_ && pBarTexture_)

303  float fCoef = (fValue_ - fMinValue_)/(fMaxValue_ - fMinValue_);
304 
305  if (mOrientation_ == orientation::HORIZONTAL)
306  {
307  pBarTexture_->set_rel_width(fCoef);
308  pBarTexture_->set_rel_height(1.0f);
309  }
310  else

312  pBarTexture_->set_rel_width(1.0f);
313  pBarTexture_->set_rel_height(fCoef);
314  }
315 
316  if (!pBarTexture_->get_texture().empty())
317  {
318  std::array<float,4> uvs = lInitialTextCoords_;
319  if (mOrientation_ == orientation::HORIZONTAL)
320  {
321  if (bReversed_)
322  uvs[0] = (uvs[0] - uvs[2])*fCoef + uvs[2];
323  else
324  uvs[2] = (uvs[2] - uvs[0])*fCoef + uvs[0];
325  }
326  else
327  {
328  if (bReversed_)
329  uvs[3] = (uvs[3] - uvs[1])*fCoef + uvs[1];
330  else
331  uvs[1] = (uvs[1] - uvs[3])*fCoef + uvs[3];
332  }
333 
334  pBarTexture_->set_tex_coord(uvs);
335  }
336 
337  bUpdateBarTexture_ = false;
338  }
339 
340  alive_checker mChecker(this);
341  frame::update(fDelta);
342  if (!mChecker.is_alive())
343  return;
