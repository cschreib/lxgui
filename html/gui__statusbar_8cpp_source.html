287 void status_bar::update(float fDelta)
288 {
289  if (bUpdateBarTexture_ && pBarTexture_)
290  {
291  float fCoef = (fValue_ - fMinValue_)/(fMaxValue_ - fMinValue_);
292 
293  if (mOrientation_ == orientation::HORIZONTAL)
294  pBarTexture_->set_relative_dimensions(vector2f(fCoef, 1.0f));
295  else
296  pBarTexture_->set_relative_dimensions(vector2f(1.0f, fCoef));
297 
298  std::array<float,4> uvs = lInitialTextCoords_;
299  if (mOrientation_ == orientation::HORIZONTAL)
300  {
301  if (bReversed_)
302  uvs[0] = (uvs[0] - uvs[2])*fCoef + uvs[2];
303  else
304  uvs[2] = (uvs[2] - uvs[0])*fCoef + uvs[0];
305  }
306  else
307  {
308  if (bReversed_)
309  uvs[3] = (uvs[3] - uvs[1])*fCoef + uvs[1];
310  else
311  uvs[1] = (uvs[1] - uvs[3])*fCoef + uvs[3];
312  }
313 
314  pBarTexture_->set_tex_rect(uvs);
315 
316  bUpdateBarTexture_ = false;
317  }
318 
319  alive_checker mChecker(*this);
320  base::update(fDelta);
321  if (!mChecker.is_alive())
322  return;
323 }
