91 void renderer::render_quads(const material* pMaterial,
92  const std::vector<std::array<vertex,4>>& lQuadList)
93 {
94  if (lQuadList.empty())
95  return;
96 
97  if (!is_quad_batching_enabled())
98  {
99  // Render immediately
100  render_quads_(pMaterial, lQuadList);
101  return;
102  }
103 
104  if (!uses_same_texture_(pMaterial, pCurrentMaterial_))
105  {
106  // Render current batch and start a new one
107  flush_quad_batch();
108  pCurrentMaterial_ = pMaterial;
109  }
110 
111  if (lQuadCache_[uiCurrentQuadCache_].lData.empty())
112  {
113  // Start a new batch
114  pCurrentMaterial_ = pMaterial;
115  }
116 
117  if (!pCurrentMaterial_)
118  {
119  // Previous quads had no material, override with the new one
120  pCurrentMaterial_ = pMaterial;
121  }
122 
123  // Add to the cache
124  auto& mCache = lQuadCache_[uiCurrentQuadCache_];
125 
126  if (!pMaterial && is_texture_atlas_enabled() && is_texture_vertex_color_supported())
127  {
128  // To allow quads with no texture to enter the batch
129  // with atlas textures, we just change their UV coordinates
130  // to map to the first top-left pixel of the atlas, which is always white.
131  mCache.lData.reserve(mCache.lData.size() + lQuadList.size());
132  for (const auto& mOrigQuad : lQuadList)
133  {
134  mCache.lData.push_back(mOrigQuad);
135  auto& mQuad = mCache.lData.back();
136  mQuad[0].uvs = mQuad[1].uvs = mQuad[2].uvs = mQuad[3].uvs = vector2f(0.0f, 0.0f);
137  }
138  }
139  else
140  {
141  mCache.lData.insert(mCache.lData.end(), lQuadList.begin(), lQuadList.end());
142  }
143 }

145 void renderer::flush_quad_batch()
146 {
147  auto& mCache = lQuadCache_[uiCurrentQuadCache_];
148  if (mCache.lData.empty())
149  return;
150 
151  if (mCache.pCache)
152  {
153  mCache.pCache->update(mCache.lData[0].data(), mCache.lData.size()*4);
154  render_cache_(pCurrentMaterial_, *mCache.pCache, matrix4f::IDENTITY);
155  }
156  else
157  {
158  render_quads_(pCurrentMaterial_, mCache.lData);
159  }
160 
161  mCache.lData.clear();
162  pCurrentMaterial_ = nullptr;
163 
164  ++uiCurrentQuadCache_;
165  if (uiCurrentQuadCache_ == BATCHING_CACHE_CYCLE_SIZE)
166  uiCurrentQuadCache_ = 0u;
167 
168  ++uiBatchCount_;
169 }
