363  } else if (borders_.right - borders_.left > max_width_) {
364  borders_.right = borders_.left + max_width_;
365  }
366 
367  if (borders_.bottom - borders_.top < min_height_) {
368  borders_.bottom = borders_.top + min_height_;
369  } else if (borders_.bottom - borders_.top > max_height_) {
370  borders_.bottom = borders_.top + max_height_;
371  }
372 
373  if (is_clamped_to_screen_) {
374  vector2f screen_dimensions = get_effective_frame_renderer()->get_target_dimensions();
375

891 std::string
892 hijack_sol_error_line(std::string original_message, const std::string& file, std::size_t line_nbr) {
893  auto pos1 = original_message.find("[string \"" + file);
894  if (pos1 == std::string::npos)
895  return original_message;
896 
897  auto pos2 = original_message.find_first_of('"', pos1 + 9);
898  if (pos2 == std::string::npos)
899  return original_message;
900 
901  original_message.erase(pos1, pos2 - pos1 + 2);
902  original_message.insert(pos1, file);
903 
904  auto pos3 = original_message.find_first_of(':', pos1 + file.size());
905  if (pos3 == std::string::npos)
906  return original_message;
907 
908  auto pos4 = original_message.find_first_of(":>", pos3 + 1);
909  if (pos4 == std::string::npos)
910  return original_message;
911 
912  auto offset =
913  utils::from_string<std::size_t>(original_message.substr(pos3 + 1, pos4 - pos3 - 1));
914  if (!offset.has_value())
915  return original_message;
916 
917  original_message.erase(pos3 + 1, pos4 - pos3 - 1);
918  original_message.insert(pos3 + 1, utils::to_string(line_nbr + offset.value() - 1));
919  pos4 = original_message.find_first_of(':', pos3 + 1);
920 
921  auto pos5 = original_message.find("[string \"" + file, pos4);
922  if (pos5 == std::string::npos)
923  return original_message;
924 
925  std::string message = original_message.substr(pos4 + 1);
926  original_message.erase(pos4 + 1);
927  original_message += hijack_sol_error_line(message, file, line_nbr);
928 
929  return original_message;
930 }

991 utils::connection frame::define_script_(
992  const std::string& script_name,
993  sol::protected_function handler,
994  bool append,
995  const script_info& info) {
996 
997  auto wrapped_handler = [handler = std::move(handler),
998  info](frame& self, const event_data& args) {
999  sol::state& lua = self.get_manager().get_lua();
1000  lua_State* lua_raw = lua.lua_state();
1001 
1002  std::vector<sol::object> lua_args;
1003 
1004  // Set arguments
1005  for (std::size_t i = 0; i < args.get_param_count(); ++i) {
1006  const utils::variant& arg = args.get(i);
1007  if (std::holds_alternative<utils::empty>(arg))
1008  lua_args.emplace_back(sol::lua_nil);
1009  else
1010  lua_args.emplace_back(lua_raw, sol::in_place, arg);
1011  }
1012 
1013  // Get a reference to self
1014  sol::object self_lua = lua[self.get_name()];
1015  if (self_lua == sol::lua_nil)
1016  throw gui::exception("Lua glue object is nil");
1017 
1018  // Call the function
1019  auto result = handler(self_lua, sol::as_args(lua_args));
1020  // WARNING: after this point, the frame (self_lua) may be deleted.
1021  // Do not use any member variable or member function directly.
1022 
1023  // Handle errors
1024  if (!result.valid()) {
1025  throw gui::exception(hijack_sol_error_message(
1026  result.get<sol::error>().what(), info.file_name, info.line_nbr));
1027  }
1028  };
1029 
1030  return define_script_(script_name, std::move(wrapped_handler), append, info);
1031 }
