21 void font_string::render() const {
22  base::render();
23 
24  if (!text_ || !is_ready_ || !is_visible())
25  return;
26 
27  text_->set_use_vertex_cache(is_vertex_cache_used_());
28 
29  vector2f pos;
30 
31  if (std::isinf(text_->get_box_width())) {
32  switch (align_x_) {
33  case alignment_x::left: pos.x = borders_.left; break;
34  case alignment_x::center: pos.x = (borders_.left + borders_.right) / 2; break;
35  case alignment_x::right: pos.x = borders_.right; break;
36  }
37  } else {
38  pos.x = borders_.left;
39  }
40 
41  if (std::isinf(text_->get_box_height())) {
42  switch (align_y_) {
43  case alignment_y::top: pos.y = borders_.top; break;
44  case alignment_y::middle: pos.y = (borders_.top + borders_.bottom) / 2; break;
45  case alignment_y::bottom: pos.y = borders_.bottom; break;
46  }
47  } else {
48  pos.y = borders_.top;
49  }
50 
51  pos += offset_;
52 
53  text_->set_alpha(get_effective_alpha());
54 
55  if (is_shadow_enabled_) {
56  text_->set_color(shadow_color_, true);
57  text_->render(matrix4f::translation(round_to_pixel(pos + shadow_offset_)));
58  }
59 
60  text_->set_color(text_color_);
61  text_->render(matrix4f::translation(round_to_pixel(pos)));
62 }

179 void font_string::create_text_object_() {
180  if (font_name_.empty())
181  return;
182 
183  std::size_t pixel_height = static_cast<std::size_t>(
184  std::round(get_manager().get_interface_scaling_factor() * height_));
185 
186  auto& renderer = get_manager().get_renderer();
187  const auto& localizer = get_manager().get_localizer();
188 
189  const auto& code_points = localizer.get_allowed_code_points();
190  const char32_t default_code_point = localizer.get_fallback_code_point();
191 
192  std::shared_ptr<gui::font> outline_font;
193  if (is_outlined_) {
194  outline_font = renderer.create_atlas_font(
195  "GUI", font_name_, pixel_height,
196  std::min<std::size_t>(2u, static_cast<std::size_t>(std::round(0.2 * pixel_height))),
197  code_points, default_code_point);
198  }
199 
200  auto fnt = renderer.create_atlas_font(
201  "GUI", font_name_, pixel_height, 0u, code_points, default_code_point);
202 
203  text_ = std::unique_ptr<text>(new text(renderer, fnt, outline_font));
204 
205  text_->set_scaling_factor(1.0f / get_manager().get_interface_scaling_factor());
206  text_->set_remove_starting_spaces(true);
207  text_->set_text(content_);
208  text_->set_alignment_x(align_x_);
209  text_->set_alignment_y(align_y_);
210  text_->set_tracking(spacing_);
211  text_->set_word_wrap_enabled(word_wrap_enabled_);
212  text_->set_word_ellipsis_enabled(ellipsis_enabled_);
213  text_->set_formatting_enabled(formatting_enabled_);

435 void font_string::update_borders_() {
436  if (!text_)
437  return base::update_borders_();
438 
439 //#define DEBUG_LOG(msg) gui::out << (msg) << std::endl
440 #define DEBUG_LOG(msg)
441 
442  const bool old_ready = is_ready_;
443  const auto old_border_list = borders_;
444  is_ready_ = true;
445 
446  if (!anchor_list_.empty()) {
447  float left = 0.0f, right = 0.0f, top = 0.0f, bottom = 0.0f;
448  float x_center = 0.0f, y_center = 0.0f;
449 
450  DEBUG_LOG(" Read anchors");
451  read_anchors_(left, right, top, bottom, x_center, y_center);
452 
453  float box_width = std::numeric_limits<float>::infinity();
454  if (get_dimensions().x != 0.0f)
455  box_width = get_dimensions().x;
456  else if (defined_borders_.left && defined_borders_.right)
457  box_width = right - left;
458 
459  float box_height = std::numeric_limits<float>::infinity();
460  if (get_dimensions().y != 0.0f)
461  box_height = get_dimensions().y;
462  else if (defined_borders_.top && defined_borders_.bottom)
463  box_height = bottom - top;
464 
465  box_width = round_to_pixel(box_width, utils::rounding_method::nearest_not_zero);
466  box_height = round_to_pixel(box_height, utils::rounding_method::nearest_not_zero);
467 
468  text_->set_box_dimensions(box_width, box_height);
469 
470  DEBUG_LOG(" Make borders");
471  if (std::isinf(box_height))
472  box_height = text_->get_height();
473  if (std::isinf(box_width))
474  box_width = text_->get_width();
475 
476  if (!make_borders_(top, bottom, y_center, box_height)) {
477  is_ready_ = false;
478  }
479 
480  if (!make_borders_(left, right, x_center, box_width)) {
481  is_ready_ = false;
482  }
483 
484  if (is_ready_) {
485  if (right < left) {
486  right = left + 1.0f;
487  }
488  if (bottom < top) {
489  bottom = top + 1.0f;
490  }
491 
492  borders_.left = left;
493  borders_.right = right;
494  borders_.top = top;
495  borders_.bottom = bottom;
496  } else {
497  borders_ = bounds2f::zero;
498  }
499  } else {
500  float box_width = get_dimensions().x;
501  if (box_width == 0.0) {
502  box_width = text_->get_width();
503  }
504 
505  float box_height = get_dimensions().y;
506  if (box_height == 0.0) {
507  box_height = text_->get_height();
508  }
509 
510  borders_ = bounds2f(0.0, 0.0, box_width, box_height);
511  is_ready_ = false;
512  }
513 
514  borders_.left = round_to_pixel(borders_.left);
515  borders_.right = round_to_pixel(borders_.right);
516  borders_.top = round_to_pixel(borders_.top);
517  borders_.bottom = round_to_pixel(borders_.bottom);
518 
519  if (borders_ != old_border_list || is_ready_ != old_ready) {
520  DEBUG_LOG(" Fire redraw");
521  notify_renderer_need_redraw();
522  }
523  DEBUG_LOG(" @");
524 }
