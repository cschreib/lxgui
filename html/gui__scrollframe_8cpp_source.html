86 void scroll_frame::set_scroll_child(utils::owner_ptr<frame> pFrame)
87 {
88  if (pScrollChild_)

92  }

98 
99  if (!pScrollTexture)
100  return;
101 
102  pScrollTexture->set_special();
103  pScrollTexture->set_all_points(observer_from(this));
104 
105  if (pScrollRenderTarget_)
106  pScrollTexture->set_texture(pScrollRenderTarget_);
107 
108  pScrollTexture->notify_loaded();
109  pScrollTexture_ = pScrollTexture;
110 
111  bRebuildScrollRenderTarget_ = true;
112  }
113 
114  pScrollChild_ = pFrame;
115 
116  if (pScrollChild_)
117  {
118  add_child(std::move(pFrame));
119 
120  pScrollChild_->set_special();
121  if (!is_virtual())
122  pScrollChild_->set_renderer(observer_from(this));
123 
124  pScrollChild_->clear_all_points();
125  pScrollChild_->set_point(anchor_point::TOP_LEFT, get_name(), -mScroll_);
126 
127  update_scroll_range_();
128  bUpdateScrollRange_ = false;
129  }
130 
131  bRedrawScrollRenderTarget_ = true;

190 void scroll_frame::update(float fDelta)
191 {
192  vector2f mOldChildSize;
193  if (pScrollChild_)
194  mOldChildSize = pScrollChild_->get_apparent_dimensions();
195 
196  alive_checker mChecker(*this);
197  base::update(fDelta);
198  if (!mChecker.is_alive())
199  return;
200 
201  if (pScrollChild_ && mOldChildSize != pScrollChild_->get_apparent_dimensions())
202  {
203  bUpdateScrollRange_ = true;
204  bRedrawScrollRenderTarget_ = true;
205  }
206 
207  if (is_visible())
208  {
209  if (bRebuildScrollRenderTarget_ && pScrollTexture_)
210  {
211  rebuild_scroll_render_target_();
212  bRebuildScrollRenderTarget_ = false;
213  bRedrawScrollRenderTarget_ = true;
214  }
215 
216  if (bUpdateScrollRange_)
217  {
218  update_scroll_range_();
219  bUpdateScrollRange_ = false;
220  }
221 
222  if (pScrollChild_ && pScrollRenderTarget_ && bRedrawScrollRenderTarget_)
223  {
224  render_scroll_strata_list_();
225  bRedrawScrollRenderTarget_ = false;
226  }

230 void scroll_frame::update_scroll_range_()
231 {
232  const vector2f mApparentSize = get_apparent_dimensions();
233  const vector2f mChildApparentSize = pScrollChild_->get_apparent_dimensions();
234 
235  mScrollRange_ = mChildApparentSize - mApparentSize;
236 
237  if (mScrollRange_.x < 0) mScrollRange_.x = 0;
238  if (mScrollRange_.y < 0) mScrollRange_.y = 0;
239 
240  if (!is_virtual())
241  {
242  alive_checker mChecker(*this);
243  fire_script("OnScrollRangeChanged");
244  if (!mChecker.is_alive())

256 void scroll_frame::rebuild_scroll_render_target_()
257 {
258  const vector2f mApparentSize = get_apparent_dimensions();
259 
260  if (mApparentSize.x <= 0 || mApparentSize.y <= 0)
261  return;
262 
263  float fFactor = get_manager().get_interface_scaling_factor();
264  vector2ui mScaledSize = vector2ui(
265  std::round(mApparentSize.x*fFactor), std::round(mApparentSize.y*fFactor));
266 
267  if (pScrollRenderTarget_)
268  {
269  pScrollRenderTarget_->set_dimensions(mScaledSize);
270  pScrollTexture_->set_tex_rect(std::array<float,4>{0.0f, 0.0f, 1.0f, 1.0f});
271  bUpdateScrollRange_ = true;
272  }
273  else
274  {
275  auto& mRenderer = get_manager().get_renderer();
276  pScrollRenderTarget_ = mRenderer.create_render_target(mScaledSize);
277 
278  if (pScrollRenderTarget_)
279  pScrollTexture_->set_texture(pScrollRenderTarget_);
280  }
281 }
