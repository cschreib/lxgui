180 void scroll_frame::update(float fDelta)
181 {
182  vector2f mOldChildSize;
183  if (pScrollChild_)
184  mOldChildSize = pScrollChild_->get_apparent_dimensions();
185 
186  alive_checker mChecker(*this);
187  base::update(fDelta);
188  if (!mChecker.is_alive())
189  return;
190 
191  if (pScrollChild_ && mOldChildSize != pScrollChild_->get_apparent_dimensions())
192  {
193  bUpdateScrollRange_ = true;
194  bRedrawScrollRenderTarget_ = true;
195  }
196 
197  if (is_visible())
198  {
199  if (bRebuildScrollRenderTarget_ && pScrollTexture_)
200  {
201  rebuild_scroll_render_target_();
202  bRebuildScrollRenderTarget_ = false;
203  bRedrawScrollRenderTarget_ = true;
204  }
205 
206  if (bUpdateScrollRange_)
207  {
208  update_scroll_range_();
209  bUpdateScrollRange_ = false;
210  }
211 
212  if (pScrollChild_ && pScrollRenderTarget_ && bRedrawScrollRenderTarget_)
213  {
214  render_scroll_strata_list_();
215  bRedrawScrollRenderTarget_ = false;
216  }
217  }

220 void scroll_frame::update_scroll_range_()
221 {
222  const vector2f mApparentSize = get_apparent_dimensions();
223  const vector2f mChildApparentSize = pScrollChild_->get_apparent_dimensions();
224 
225  mScrollRange_ = mChildApparentSize - mApparentSize;
226 
227  if (mScrollRange_.x < 0) mScrollRange_.x = 0;
228  if (mScrollRange_.y < 0) mScrollRange_.y = 0;
229 
230  if (!is_virtual())
231  {
232  alive_checker mChecker(*this);
233  fire_script("OnScrollRangeChanged");
234  if (!mChecker.is_alive())
235  return;
236  }

246 void scroll_frame::rebuild_scroll_render_target_()
247 {
248  const vector2f mApparentSize = get_apparent_dimensions();
249 
250  if (mApparentSize.x <= 0 || mApparentSize.y <= 0)
251  return;
252 
253  float fFactor = get_manager().get_interface_scaling_factor();
254  vector2ui mScaledSize = vector2ui(
255  std::round(mApparentSize.x*fFactor), std::round(mApparentSize.y*fFactor));
256 
257  if (pScrollRenderTarget_)
258  {
259  pScrollRenderTarget_->set_dimensions(mScaledSize);
260  pScrollTexture_->set_tex_rect(std::array<float,4>{0.0f, 0.0f, 1.0f, 1.0f});
261  bUpdateScrollRange_ = true;
262  }
263  else
264  {
265  auto& mRenderer = get_manager().get_renderer();
266  pScrollRenderTarget_ = mRenderer.create_render_target(mScaledSize);
267 
268  if (pScrollRenderTarget_)
269  pScrollTexture_->set_texture(pScrollRenderTarget_);
270  }
271 }
