77 void frame::parse_resize_bounds_node_(const layout_node& node) {
78  if (const layout_node* resize_bounds_node = node.try_get_child("ResizeBounds")) {
79  if (const layout_node* min_node = resize_bounds_node->try_get_child("Min")) {
80  auto dimensions = parse_dimension_(*min_node);
81  bool has_x = dimensions.second.x.has_value();
82  bool has_y = dimensions.second.y.has_value();
83  if (dimensions.first == anchor_type::abs) {
84  if (has_x && has_y) {
85  set_min_dimensions(
86  vector2f(dimensions.second.x.value(), dimensions.second.y.value()));
87  } else if (has_x)
88  set_min_width(dimensions.second.x.value());
89  else if (has_y)
90  set_min_height(dimensions.second.y.value());
91  } else {
92  gui::out << gui::warning << min_node->get_location() << ": "
93  << "\"RelDimension\" for ResizeBounds:Min is not yet supported. Skipped."
94  << std::endl;
95  }
96  }
97 
98  if (const layout_node* max_node = resize_bounds_node->try_get_child("Max")) {
99  auto dimensions = parse_dimension_(*max_node);
100  bool has_x = dimensions.second.x.has_value();

301 utils::observer_ptr<layered_region> frame::parse_region_(
302  const layout_node& node, const std::string& layer_name, const std::string& type) {
303  try {
304  auto attr = parse_core_attributes(
305  get_manager().get_root().get_registry(),
306  get_manager().get_virtual_root().get_registry(), node, observer_from(this));
307 
308  if (!type.empty())
309  attr.object_type = type;
310 
311  auto layer_id = utils::from_string<layer>(layer_name);
312  if (!layer_id.has_value()) {
313  gui::out << gui::warning << node.get_location() << ": "
314  << "Unknown layer type: \"" << layer_name << "\". Using \"ARTWORK\"."
315  << std::endl;
316  }
317 
318  auto reg = create_layered_region(layer_id.value_or(layer::artwork), attr);
319  if (!reg)
320  return nullptr;
321 
322  try {
323  reg->parse_layout(node);
324  reg->notify_loaded();
325  return reg;
326  } catch (...) {
327  reg->release_from_parent();
328  throw;
329  }
330  } catch (const exception& e) {
331  gui::out << gui::error << e.get_description() << std::endl;
332  }
333 
334  return nullptr;
