31 if (color_node && texture_node) {

43 if (bar_texture) {

45 set_bar_texture(utils::static_pointer_cast<texture>(bar_texture));
46 }

48 warn_for_not_accessed_node(defaulted);
49 texture_node->bypass_access_check();
50 } else if (color_node) {
