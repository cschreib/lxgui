22 void font_string::render() const
23 {
24  if (!pText_ || !bReady_ || !is_visible())
25  return;
26 
27  float fX = 0.0f, fY = 0.0f;
28 
29  if (std::isinf(pText_->get_box_width()))
30  {
31  switch (mAlignX_)
32  {
33  case alignment_x::LEFT : fX = lBorderList_.left; break;
34  case alignment_x::CENTER : fX = (lBorderList_.left + lBorderList_.right)/2; break;
35  case alignment_x::RIGHT : fX = lBorderList_.right; break;
36  }
37  }
38  else
39  fX = lBorderList_.left;
40 
41  if (std::isinf(pText_->get_box_height()))
42  {
43  switch (mAlignY_)
44  {
45  case alignment_y::TOP : fY = lBorderList_.top; break;
46  case alignment_y::MIDDLE : fY = (lBorderList_.top + lBorderList_.bottom)/2; break;
47  case alignment_y::BOTTOM : fY = lBorderList_.bottom; break;
48  }
49  }
50  else
51  fY = lBorderList_.top;
52 
53  fX += mOffset_.x;
54  fY += mOffset_.y;
55 
56  pText_->set_alpha(get_effective_alpha());
57 
58  if (bHasShadow_)
59  {
60  pText_->set_color(mShadowColor_, true);
61  pText_->render(matrix4f::translation(
62  round_to_pixel(vector2f(fX, fY) + mShadowOffset_)));
63  }
64 
65  pText_->set_color(mTextColor_);
66  pText_->render(matrix4f::translation(round_to_pixel(vector2f(fX, fY))));
67 }

232 
233  std::shared_ptr<gui::font> pOutlineFont;
234  if (bIsOutlined_)
235  {
236  pOutlineFont = mRenderer.create_atlas_font(
237  "GUI", sFontName_, uiPixelHeight,
238  std::min<std::size_t>(2u, static_cast<std::size_t>(std::round(0.2*uiPixelHeight))),
239  lCodePoints, uiDefaultCodePoint);
240  }
241 
242  auto pFont = mRenderer.create_atlas_font(
243  "GUI", sFontName_, uiPixelHeight,
244  0u, lCodePoints, uiDefaultCodePoint);
245

247 
248  pText_->set_scaling_factor(1.0f/get_manager().get_interface_scaling_factor());
249  pText_->set_remove_starting_spaces(true);

478 void font_string::update_borders_()
479 {
480  if (!pText_)
481  return base::update_borders_();
482 
483  //#define DEBUG_LOG(msg) gui::out << (msg) << std::endl
484  #define DEBUG_LOG(msg)
485 
486  const bool bOldReady = bReady_;
487  const auto lOldBorderList = lBorderList_;
488  bReady_ = true;
489 
490  if (!lAnchorList_.empty())
491  {
492  float fLeft = 0.0f, fRight = 0.0f, fTop = 0.0f, fBottom = 0.0f;
493  float fXCenter = 0.0f, fYCenter = 0.0f;
494 
495  DEBUG_LOG(" Read anchors");
496  read_anchors_(fLeft, fRight, fTop, fBottom, fXCenter, fYCenter);
497 
498  float fBoxWidth = std::numeric_limits<float>::infinity();
499  if (get_dimensions().x != 0.0f)
500  fBoxWidth = get_dimensions().x;
501  else if (lDefinedBorderList_.left && lDefinedBorderList_.right)
502  fBoxWidth = fRight - fLeft;
503 
504  float fBoxHeight = std::numeric_limits<float>::infinity();
505  if (get_dimensions().y != 0.0f)
506  fBoxHeight = get_dimensions().y;
507  else if (lDefinedBorderList_.top && lDefinedBorderList_.bottom)
508  fBoxHeight = fBottom - fTop;
509 
510  fBoxWidth = round_to_pixel(fBoxWidth, utils::rounding_method::NEAREST_NOT_ZERO);
511  fBoxHeight = round_to_pixel(fBoxHeight, utils::rounding_method::NEAREST_NOT_ZERO);
512 
513  pText_->set_dimensions(fBoxWidth, fBoxHeight);
514 
515  DEBUG_LOG(" Make borders");
516  if (std::isinf(fBoxHeight))
517  fBoxHeight = pText_->get_height();
518  if (std::isinf(fBoxWidth))
519  fBoxWidth = pText_->get_width();
520 
521  if (!make_borders_(fTop, fBottom, fYCenter, fBoxHeight))
522  bReady_ = false;
523  if (!make_borders_(fLeft, fRight, fXCenter, fBoxWidth))
524  bReady_ = false;
525 
526  if (bReady_)
527  {
528  if (fRight < fLeft)
529  fRight = fLeft + 1.0f;
530  if (fBottom < fTop)
531  fBottom = fTop + 1.0f;
532 
533  lBorderList_.left = fLeft;
534  lBorderList_.right = fRight;
535  lBorderList_.top = fTop;
536  lBorderList_.bottom = fBottom;
537  }
538  else
539  lBorderList_ = bounds2f::ZERO;
540  }
541  else
542  {
543  float fBoxWidth = get_dimensions().x;
544  if (fBoxWidth == 0.0)
545  fBoxWidth = pText_->get_width();
546 
547  float fBoxHeight = get_dimensions().y;
548  if (fBoxHeight == 0.0)
549  fBoxHeight = pText_->get_height();
550 
551  lBorderList_ = bounds2f(0.0, 0.0, fBoxWidth, fBoxHeight);
552  bReady_ = false;
553  }
554 
555  lBorderList_.left = round_to_pixel(lBorderList_.left);
556  lBorderList_.right = round_to_pixel(lBorderList_.right);
557  lBorderList_.top = round_to_pixel(lBorderList_.top);
558  lBorderList_.bottom = round_to_pixel(lBorderList_.bottom);
559 
560  if (lBorderList_ != lOldBorderList || bReady_ != bOldReady)
561  {
562  DEBUG_LOG(" Fire redraw");
563  notify_renderer_need_redraw();
564  }
565  DEBUG_LOG(" @");
