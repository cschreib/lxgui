34 void addon_registry::load_addon_toc_(const std::string& sAddOnName, const std::string& sAddOnDirectory)
35 {
36  auto& lAddOns = lAddOnList_[sAddOnDirectory];
37  if (lAddOns.find(sAddOnName) != lAddOns.end())
38  return;
39 
40  addon mAddOn;
41  mAddOn.bEnabled = true;
42  mAddOn.sMainDirectory = utils::cut(sAddOnDirectory, "/").back();
43  mAddOn.sDirectory = sAddOnDirectory + "/" + sAddOnName;
44 
45  std::string sTOCFile = mAddOn.sDirectory + "/" + sAddOnName + ".toc";
46  std::ifstream mFile(sTOCFile);
47  if (!mFile.is_open())
48  return;
49 
50  std::string sLine;
51  while (std::getline(mFile, sLine))
52  {
53  utils::replace(sLine, "\r", "");
54  if (sLine.empty())
55  continue;
56 
57  std::string_view sLineView = sLine;
58 
59  if (sLineView.size() >= 2 && sLineView[0] == '#' && sLineView[1] == '#')
60  {
61  sLineView = sLineView.substr(2);
62  sLineView = utils::trim(sLineView, ' ');
63  auto lArgs = utils::cut_first(sLineView, ":");
64  if (!lArgs.first.empty() && !lArgs.second.empty())
65  {
66  std::string_view sKey = utils::trim(lArgs.first, ' ');
67  std::string_view sValue = utils::trim(lArgs.second, ' ');
68 
69  if (sKey == "Interface")
70  {
71  mAddOn.sUIVersion = sValue;
72 
73  if (mAddOn.sUIVersion == LXGUI_UI_VERSION)
74  mAddOn.bEnabled = true;
75  else
76  {
77  gui::out << gui::warning << "gui::manager : "
78  << "Wrong UI version for \"" << sAddOnName << "\" (got : "
79  << mAddOn.sUIVersion << ", expected : " << LXGUI_UI_VERSION
80  << "). AddOn disabled." << std::endl;
81  mAddOn.bEnabled = false;
82  }
83  }
84  else if (sKey == "Title")
85  mAddOn.sName = sValue;
86  else if (sKey == "Version")
87  mAddOn.sVersion = sValue;
88  else if (sKey == "Author")
89  mAddOn.sAuthor = sValue;
90  else if (sKey == "SavedVariables")
91  {
92  for (auto sVar : utils::cut(sValue, ","))
93  {
94  sVar = utils::trim(sVar, ' ');
95  if (!utils::has_no_content(sVar))
96  mAddOn.lSavedVariableList.push_back(std::string{sVar});
97  }
98  }
99  }
100  }
101  else
102  {
103  sLineView = utils::trim(sLineView, ' ');
104  if (!utils::has_no_content(sLineView))
105  mAddOn.lFileList.push_back(mAddOn.sDirectory + "/" + std::string{sLineView});
106  }
107  }
108 
109  if (mAddOn.sName.empty())
110  gui::out << gui::error << "gui::manager : Missing addon name in " << sTOCFile << "." << std::endl;
111  else
112  lAddOns[sAddOnName] = mAddOn;
113 }

164 void addon_registry::load_addon_directory(const std::string& sDirectory)
165 {
166  for (const auto& sSubDir : utils::get_directory_list(sDirectory))
167  this->load_addon_toc_(sSubDir, sDirectory);
168 
169  std::vector<addon*> lCoreAddOnStack;
170  std::vector<addon*> lAddOnStack;
171  bool bCore = false;
172 
173  auto& lAddOns = lAddOnList_[sDirectory];
174 
175  std::ifstream mFile(sDirectory + "/addons.txt");
176  if (mFile.is_open())
177  {
178  std::string sLine;
179  while (std::getline(mFile, sLine))
180  {
181  utils::replace(sLine, "\r", "");
182  if (sLine.empty())
183  continue;
184 
185  std::string_view sLineView = sLine;
186 
187  if (sLineView[0] == '#')
188  {
189  sLineView = sLineView.substr(1);
190  sLineView = utils::trim(sLineView, ' ');
191  bCore = sLineView == "Core";
192  }
193  else
194  {
195  auto lArgs = utils::cut_first(sLineView, ":");
196  if (!lArgs.first.empty() && !lArgs.second.empty())
197  {
198  std::string_view sKey = utils::trim(lArgs.first, ' ');
199  std::string_view sValue = utils::trim(lArgs.second, ' ');
200  auto iter = lAddOns.find(std::string{sKey});
201  if (iter != lAddOns.end())
202  {
203  if (bCore)
204  lCoreAddOnStack.push_back(&iter->second);
205  else
206  lAddOnStack.push_back(&iter->second);
207 
208  iter->second.bEnabled = sValue == "1";
209  }
210  }
211  }
212  }
213  mFile.close();
214  }
215 
216  for (auto* pAddOn : lCoreAddOnStack)
217  {
218  if (pAddOn->bEnabled)
219  this->load_addon_files_(*pAddOn);
220  }
221 
222  for (auto* pAddOn : lAddOnStack)
223  {
224  if (pAddOn->bEnabled)
225  this->load_addon_files_(*pAddOn);
226  }
227 
228  pCurrentAddOn_ = nullptr;
229 }

231 std::string serialize(const std::string& sTab, const sol::object& mValue) noexcept
232 {
233  if (mValue.is<double>())
234  {
235  return utils::to_string(mValue.as<double>());
236  }
237  else if (mValue.is<int>())
238  {
239  return utils::to_string(mValue.as<int>());
240  }
241  else if (mValue.is<std::string>())
242  {
243  return "\"" + utils::to_string(mValue.as<std::string>()) + "\"";
244  }
245  else if (mValue.is<sol::table>())
246  {
247  std::string sResult;
248  sResult += "{";
249 
250  std::string sContent;
251  sol::table mTable = mValue.as<sol::table>();
252  for (const auto& mKeyValue : mTable)
253  {
254  sContent += sTab + " [" + serialize("", mKeyValue.first) + "] = "
255  + serialize(sTab + " ", mKeyValue.second) + ",\n";
256  }
257 
258  if (!sContent.empty())
259  sResult += "\n" + sContent + sTab;
260 
261  sResult += "}";
262  return sResult;
263  }
264 
265  return "nil";
266 }
