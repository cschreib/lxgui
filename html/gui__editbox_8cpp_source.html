122 void edit_box::fire_script(const std::string& sScriptName, const event_data& mData)
123 {
124  alive_checker mChecker(*this);
125 
126  // Do not fire OnKeyUp/OnKeyDown events when typing
127  bool bBypassEvent = false;
128  if (has_focus() && (sScriptName == "OnKeyUp" || sScriptName == "OnKeyDown"))
129  bBypassEvent = true;
130  if (!has_focus() && (sScriptName == "OnChar"))
131  bBypassEvent = true;
132 
133  if (!bBypassEvent)
134  {
135  base::fire_script(sScriptName, mData);
136  if (!mChecker.is_alive())
137  return;
138  }
139 
140  if (sScriptName == "OnKeyDown" && has_focus())
141  {
142  key mKey = mData.get<key>(0);
143  bool bShiftIsPressed = mData.get<bool>(2);
144  bool bCtrlIsPressed = mData.get<bool>(3);
145 
146  if (mKey == key::K_RETURN || mKey == key::K_NUMPADENTER)
147  {
148  fire_script("OnEnterPressed");
149  if (!mChecker.is_alive())
150  return;
151  }
152  else if (mKey == key::K_TAB)
153  {
154  fire_script("OnTabPressed");
155  if (!mChecker.is_alive())
156  return;
157  }
158  else if (mKey == key::K_UP)
159  {
160  fire_script("OnUpPressed");
161  if (!mChecker.is_alive())
162  return;
163  }
164  else if (mKey == key::K_DOWN)
165  {
166  fire_script("OnDownPressed");
167  if (!mChecker.is_alive())
168  return;
169  }
170  else if (mKey == key::K_SPACE)
171  {
172  fire_script("OnSpacePressed");
173  if (!mChecker.is_alive())
174  return;
175  }
176  else if (mKey == key::K_ESCAPE)
177  {
178  fire_script("OnEscapePressed");
179  if (!mChecker.is_alive())
180  return;
181  }
182 
183  process_key_(mKey, bShiftIsPressed, bCtrlIsPressed);
184 
185  if (!mChecker.is_alive())
186  return;
187  }
188  else if (sScriptName == "OnChar" && has_focus())
189  {
190  std::uint32_t c = mData.get<std::uint32_t>(1);
191  if (add_char_(c))
192  {
193  fire_script("OnTextChanged");
194  if (!mChecker.is_alive())
195  return;
196  }
197  }
198  else if (sScriptName == "OnSizeChanged")
199  {
200  update_displayed_text_();
201  update_font_string_();
202  update_carret_position_();
203  }
204  else if (sScriptName == "OnDragStart")
205  {
206  uiSelectionEndPos_ = uiSelectionStartPos_ = get_letter_id_at_(
207  vector2f(mData.get<float>(1), mData.get<float>(2)));
208  }
209  else if (sScriptName == "OnDragMove")
210  {
211  std::size_t uiPos = get_letter_id_at_(vector2f(mData.get<float>(0), mData.get<float>(1)));
212  if (uiPos != uiSelectionEndPos_)
213  {
214  if (uiPos != std::numeric_limits<std::size_t>::max())
215  {
216  highlight_text(uiSelectionStartPos_, uiPos);
217  iterCarretPos_ = sUnicodeText_.begin() + uiPos;
218  update_carret_position_();
219  }
220  else
221  {
222  std::size_t uiTemp = uiSelectionStartPos_;
223  unlight_text();
224  uiSelectionStartPos_ = uiTemp;
225  iterCarretPos_ = sUnicodeText_.begin() + uiSelectionStartPos_;
226  update_carret_position_();
227  }
228  }
229  }
230  else if (sScriptName == "OnMouseDown")
231  {
232  set_focus(true);
233  if (!mChecker.is_alive())
234  return;
235 
236  unlight_text();
237 
238  move_carret_at_({mData.get<float>(1), mData.get<float>(2)});
239  }
