18 scroll_frame::~scroll_frame() {
19  // Make sure the scroll child is destroyed now.
20  // It relies on this scroll_frame still being alive
21  // when being destroyed, but the scroll_frame destructor
22  // would be called before its inherited frame destructor
23  // (which would otherwise take care of destroying the scroll child).
24  if (scroll_child_)
25  remove_child(scroll_child_);
26 }

165 void scroll_frame::update(float delta) {
166  vector2f old_child_size;
167  if (scroll_child_)
168  old_child_size = scroll_child_->get_apparent_dimensions();
169 
170  alive_checker checker(*this);
171  base::update(delta);
172  if (!checker.is_alive())
173  return;
174 
175  if (scroll_child_ && old_child_size != scroll_child_->get_apparent_dimensions()) {
176  update_scroll_range_flag_ = true;
177  redraw_scroll_render_target_flag_ = true;
178  }
179 
180  if (is_visible()) {
181  if (rebuild_scroll_render_target_flag_ && scroll_texture_) {
182  rebuild_scroll_render_target_();
183  rebuild_scroll_render_target_flag_ = false;
184  redraw_scroll_render_target_flag_ = true;
185  }
186 
187  if (update_scroll_range_flag_) {
188  update_scroll_range_();
189  update_scroll_range_flag_ = false;
190  }
191 
192  if (scroll_child_ && scroll_render_target_ && redraw_scroll_render_target_flag_) {
193  render_scroll_strata_list_();
194  redraw_scroll_render_target_flag_ = false;
195  }
196  }

199 void scroll_frame::update_scroll_range_() {
200  const vector2f apparent_size = get_apparent_dimensions();
201  const vector2f child_apparent_size = scroll_child_->get_apparent_dimensions();
202 
203  scroll_range_ = child_apparent_size - apparent_size;
204 
205  if (scroll_range_.x < 0)
206  scroll_range_.x = 0;
207  if (scroll_range_.y < 0)
208  scroll_range_.y = 0;
209 
210  if (!is_virtual()) {
211  alive_checker checker(*this);
212  fire_script("OnScrollRangeChanged");
213  if (!checker.is_alive())
214  return;
215  }

224 void scroll_frame::rebuild_scroll_render_target_() {
225  const vector2f apparent_size = get_apparent_dimensions();
226 
227  if (apparent_size.x <= 0 || apparent_size.y <= 0)
228  return;
229 
230  float factor = get_manager().get_interface_scaling_factor();
231  vector2ui scaled_size =
232  vector2ui(std::round(apparent_size.x * factor), std::round(apparent_size.y * factor));
233 
234  if (scroll_render_target_) {
235  scroll_render_target_->set_dimensions(scaled_size);
236  scroll_texture_->set_tex_rect(std::array<float, 4>{0.0f, 0.0f, 1.0f, 1.0f});
237  update_scroll_range_flag_ = true;
238  } else {
239  auto& renderer = get_manager().get_renderer();
240  scroll_render_target_ = renderer.create_render_target(scaled_size);
241 
242  if (scroll_render_target_)
243  scroll_texture_->set_texture(scroll_render_target_);
244  }
245 }
