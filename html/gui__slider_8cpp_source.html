62 void slider::fire_script(const std::string& sScriptName, const event_data& mData)
63 {
64  alive_checker mChecker(*this);
65  base::fire_script(sScriptName, mData);
66  if (!mChecker.is_alive())
67  return;
68 
69  if (sScriptName == "OnDragStart")
70  {
71  if (pThumbTexture_ && pThumbTexture_->is_in_region({mData.get<float>(1), mData.get<float>(2)}))
72  {
73  anchor& mAnchor = pThumbTexture_->modify_point(anchor_point::CENTER);
74 
75  get_manager().get_root().start_moving(
76  pThumbTexture_, &mAnchor,
77  mOrientation_ == orientation::HORIZONTAL ? constraint::X : constraint::Y,
78  [&]() { constrain_thumb_(); }
79  );
80 
81  bThumbMoved_ = true;
82  }
83  }
84  else if (sScriptName == "OnDragStop")
85  {
86  if (pThumbTexture_)
87  {
88  if (get_manager().get_root().is_moving(*pThumbTexture_))
89  get_manager().get_root().stop_moving();
90 
91  bThumbMoved_ = false;
92  }
93  }
94  else if (sScriptName == "OnMouseDown")
95  {
96  if (bAllowClicksOutsideThumb_)
97  {
98  const vector2f mApparentSize = get_apparent_dimensions();
99 
100  float fValue;
101  if (mOrientation_ == orientation::HORIZONTAL)
102  {
103  float fOffset = mData.get<float>(1) - lBorderList_.left;
104  fValue = fOffset/mApparentSize.x;
105  set_value(fValue*(fMaxValue_ - fMinValue_) + fMinValue_);
106  }
107  else
108  {
109  float fOffset = mData.get<float>(2) - lBorderList_.top;
110  fValue = fOffset/mApparentSize.y;
111  set_value(fValue*(fMaxValue_ - fMinValue_) + fMinValue_);
112  }
113  }
114  }

150 void slider::constrain_thumb_()
151 {
152  if (fMaxValue_ == fMinValue_)
153  return;
154 
155  const vector2f mApparentSize = get_apparent_dimensions();
156 
157  if ((mOrientation_ == orientation::HORIZONTAL && mApparentSize.x <= 0) ||
158  (mOrientation_ == orientation::VERTICAL && mApparentSize.y <= 0))
159  return;
160 
161  float fOldValue = fValue_;
162 
163  if (bThumbMoved_)
164  {
165  if (mOrientation_ == orientation::HORIZONTAL)
166  fValue_ = pThumbTexture_->get_point(anchor_point::CENTER).mOffset.x/mApparentSize.x;
167  else
168  fValue_ = pThumbTexture_->get_point(anchor_point::CENTER).mOffset.y/mApparentSize.y;
169 
170  fValue_ = fValue_ * (fMaxValue_ - fMinValue_) + fMinValue_;
171  fValue_ = std::clamp(fValue_, fMinValue_, fMaxValue_);
172  step_value(fValue_, fValueStep_);
173  }
174 
175  float fCoef = (fValue_ - fMinValue_)/(fMaxValue_ - fMinValue_);
176 
177  anchor& mAnchor = pThumbTexture_->modify_point(anchor_point::CENTER);
178 
179  vector2f mNewOffset;
180  if (mOrientation_ == orientation::HORIZONTAL)
181  mNewOffset = vector2f(mApparentSize.x*fCoef, 0);
182  else
183  mNewOffset = vector2f(0, mApparentSize.y*fCoef);
184 
185  if (mNewOffset != mAnchor.mOffset)
186  {
187  mAnchor.mOffset = mNewOffset;
188  pThumbTexture_->notify_borders_need_update();
189  }
190 
191  if (fValue_ != fOldValue)
192  fire_script("OnValueChanged");
193 }
