62 void slider::fire_script(const std::string& sScriptName, const event_data& mData)
63 {
64  alive_checker mChecker(*this);
65  base::fire_script(sScriptName, mData);
66  if (!mChecker.is_alive())
67  return;
68 
69  if (sScriptName == "OnDragStart")
70  {
71  if (pThumbTexture_ && pThumbTexture_->is_in_region({mData.get<float>(1), mData.get<float>(2)}))
72  {
73  anchor& mAnchor = pThumbTexture_->modify_point(anchor_point::CENTER);
74 
75  get_manager().get_root().start_moving(
76  pThumbTexture_, &mAnchor,
77  mOrientation_ == orientation::HORIZONTAL ? constraint::X : constraint::Y,
78  [&]() { constrain_thumb_(); }
79  );
80 
81  bThumbMoved_ = true;
82  }
83  }
84  else if (sScriptName == "OnDragStop")
85  {
86  if (pThumbTexture_)
87  {
88  if (get_manager().get_root().is_moving(*pThumbTexture_))
89  get_manager().get_root().stop_moving();
90 
91  bThumbMoved_ = false;
92  }
93  }
94  else if (sScriptName == "OnMouseDown")
95  {
96  if (bAllowClicksOutsideThumb_)
97  {
98  const vector2f mApparentSize = get_apparent_dimensions();
99 
100  float fValue;
101  if (mOrientation_ == orientation::HORIZONTAL)
102  {
103  float fOffset = mData.get<float>(1) - lBorderList_.left;
104  fValue = fOffset/mApparentSize.x;
105  set_value(fValue*(fMaxValue_ - fMinValue_) + fMinValue_);
106  }
107  else
108  {
109  float fOffset = mData.get<float>(2) - lBorderList_.top;
110  fValue = fOffset/mApparentSize.y;
111  set_value(fValue*(fMaxValue_ - fMinValue_) + fMinValue_);
112  }
113  }
114  }

151 void slider::constrain_thumb_()
152 {
153  if (fMaxValue_ == fMinValue_)
154  return;
155 
156  const vector2f mApparentSize = get_apparent_dimensions();
157 
158  if ((mOrientation_ == orientation::HORIZONTAL && mApparentSize.x <= 0) ||
159  (mOrientation_ == orientation::VERTICAL && mApparentSize.y <= 0))
160  return;
161 
162  float fOldValue = fValue_;
163 
164  if (bThumbMoved_)
165  {
166  if (mOrientation_ == orientation::HORIZONTAL)
167  fValue_ = pThumbTexture_->get_point(anchor_point::CENTER).mOffset.x/mApparentSize.x;
168  else
169  fValue_ = pThumbTexture_->get_point(anchor_point::CENTER).mOffset.y/mApparentSize.y;
170 
171  fValue_ = fValue_ * (fMaxValue_ - fMinValue_) + fMinValue_;
172  fValue_ = std::clamp(fValue_, fMinValue_, fMaxValue_);
173  step_value(fValue_, fValueStep_);
174  }
175 
176  float fCoef = (fValue_ - fMinValue_)/(fMaxValue_ - fMinValue_);
177 
178  anchor& mAnchor = pThumbTexture_->modify_point(anchor_point::CENTER);
179 
180  vector2f mNewOffset;
181  if (mOrientation_ == orientation::HORIZONTAL)
182  mNewOffset = vector2f(mApparentSize.x*fCoef, 0);
183  else
184  mNewOffset = vector2f(0, mApparentSize.y*fCoef);
185 
186  if (mNewOffset != mAnchor.mOffset)
187  {
188  mAnchor.mOffset = mNewOffset;
189  pThumbTexture_->notify_borders_need_update();
190  }
191 
192  if (fValue_ != fOldValue)
193  fire_script("OnValueChanged");
194 }
