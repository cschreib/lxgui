229  std::string localize(std::string_view sKey, Args&& ... mArgs) const
230  {
231  if (!is_key_valid_(sKey)) return std::string{sKey};
232 
233  auto mIter = find_key_(sKey);
234  if (mIter == lMap_.end()) return std::string{sKey};
235 
236  return std::visit([&](const auto& mItem)
237  {
238  constexpr bool bIsString = std::is_same_v<std::decay_t<decltype(mItem)>, std::string>;
239  if constexpr (bIsString)
240  {
241  if constexpr (sizeof...(Args) == 0)
242  return mItem;
243  else
244  return fmt::format(mLocale_, mItem, std::forward<Args>(mArgs)...);
245  }
246  else
247  {
248  auto mResult = mItem(std::forward<Args>(mArgs)...);
249  if (mResult.valid() && mResult.begin() != mResult.end())
250  {
251  auto&& mFirst = *mResult.begin();
252  if (mFirst.template is<std::string>())
253  return mFirst.template as<std::string>();
254  else
255  return std::string{sKey};
256  } else
