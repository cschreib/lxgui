57  std::pair<std::size_t,std::size_t> get_line_info(std::size_t uiOffset) const
58  {
59  auto mIter = std::lower_bound(lLineOffsets_.begin(), lLineOffsets_.end(), uiOffset);
60  if (mIter == lLineOffsets_.end())
61  return std::make_pair(0, 0);
62 
63  std::size_t uiLineNbr = mIter - lLineOffsets_.begin();
64  std::size_t uiCharOffset = uiOffset - *mIter + 1u;
65 
66  return std::make_pair(uiLineNbr, uiCharOffset);
67  }
68 
69  std::string get_location(std::size_t uiOffset) const
70  {

72  if (mLocation.first == 0)
73  return sFileName_ + ":?";
74  else

86 std::string normalize_node_name(const std::string& sName, bool bCapitalFirst)
87 {
88  std::string sNormalized;
89  bool bNextCapitalize = bCapitalFirst;
90  for (auto cChar : sName)
91  {
92  if (bNextCapitalize)
93  cChar = std::toupper(cChar);
94 
95  bNextCapitalize = cChar == '_';
96  if (bNextCapitalize)
97  continue;
98 
99  sNormalized.push_back(cChar);
100  }
101 
102  return sNormalized;

156 void set_node(const file_line_mappings& mFile, const ryml::Tree& mTree,
157  layout_node& mNode, const ryml::NodeRef& mYAMLNode)
158 {
159  std::string sLocation;
160  if (mYAMLNode.has_key())
161  sLocation = mFile.get_location(mYAMLNode.key().data() - mTree.arena().data());
162  else if (mYAMLNode.has_val())
163  sLocation = mFile.get_location(mYAMLNode.val().data() - mTree.arena().data());
164  mNode.set_location(sLocation);
165  mNode.set_value_location(sLocation);
166 
167  if (mYAMLNode.has_key())
168  mNode.set_name(normalize_node_name(to_string(mYAMLNode.key()), true));
169 
170  for (const auto& mElemNode : mYAMLNode.children())
171  {
172  switch (mElemNode.type())
173  {
174  case ryml::KEYVAL:
175  {
176  std::string sName = normalize_node_name(to_string(mElemNode.key()), false);
177  std::string sAttrLocation = mFile.get_location(mElemNode.key().data() - mTree.arena().data());
178  if (const auto* pNode = mNode.try_get_attribute(sName))
179  {
180  gui::out << gui::warning << sAttrLocation << " : attribute '" <<
181  sName << "' duplicated; only first value will be used." << std::endl;
182  gui::out << gui::warning << std::string(sAttrLocation.size(), ' ') <<
183  " first occurence at: '" << std::endl;
184  gui::out << gui::warning << std::string(sAttrLocation.size(), ' ') <<
185  " " << pNode->get_location() << std::endl;
186  pNode->mark_as_not_accessed();
187  continue;
188  }
189 
190  auto& mAttrib = mNode.add_attribute();
191  mAttrib.set_location(std::move(sAttrLocation));
192  mAttrib.set_value_location(mFile.get_location(mElemNode.val().data() - mTree.arena().data()));
193  mAttrib.set_name(std::move(sName));
194  mAttrib.set_value(to_string(mElemNode.val()));
195  break;
196  }
197  case ryml::KEYMAP: [[fallthrough]];
198  case ryml::MAP: [[fallthrough]];
199  case ryml::KEYSEQ:
200  {
201  auto& mChild = mNode.add_child();
202  set_node(mFile, mTree, mChild, mElemNode);
203  break;
204  }
205  default:
206  {
207  gui::out << gui::warning << sLocation << " : unsupported YAML node type: '" <<
208  mElemNode.type_str() << "'." << std::endl;
209  break;
210  }
211  }

229

234

245

247  bParsed = true;

255  set_node(mFile, mTree, mRoot, mTree.rootref().first_child());

257  }
258 #endif
259 
260  if (!bParsed)
261  {
262  gui::out << gui::error << sFile << ": no parser registered for extension '" +
263  sExtension + "'." << std::endl;
264  return;
265  }
266 
267  for (const auto& mNode : mRoot.get_children())
268  {
269  if (mNode.get_name() == "Script")
270  {

272  mNode.get_attribute_value<std::string>("file");

277  }

281 
282  gui::out << gui::error << sError << std::endl;
283 
284  mEventEmitter_.fire_event("LUA_ERROR", {sError});
285  }
286  }
287  else if (mNode.get_name() == "Include")
288  {
289  parse_layout_file_(mAddOn.sDirectory+"/"+
290  mNode.get_attribute_value<std::string>("file"), mAddOn);
291  }
292  else
293  {
294  try
295  {
296  auto mAttr = parse_core_attributes(
297  mRoot_.get_registry(), mVirtualRoot_.get_registry(), mNode, nullptr);
298 
299  utils::observer_ptr<frame> pFrame;
300  auto pParent = mAttr.pParent; // copy here to prevent use-after-move
301  if (pParent)
302  {
303  pFrame = pParent->create_child(std::move(mAttr));
304  }
305  else
306  {
307  if (mAttr.bVirtual)
308  pFrame = mVirtualRoot_.create_root_frame(std::move(mAttr));
309  else
310  pFrame = mRoot_.create_root_frame(std::move(mAttr));
311  }
312

314  continue;
315

317  pFrame->parse_layout(mNode);

323  }
324  }
