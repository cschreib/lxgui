27  explicit file_line_mappings(const std::string& file_name) : file_name_(file_name) {
28  std::ifstream stream(file_name);
29  if (!stream.is_open())
30  return;
31 
32  std::string line;
33  std::size_t prev_pos = 0u;
34  while (std::getline(stream, line)) {
35  file_content_ += '\n' + line;
36  line_offsets_.push_back(prev_pos);
37  prev_pos += line.size() + 1u;
38  }
39 
40  line_offsets_.push_back(prev_pos);
41 
42  is_open_ = true;
43  }

47  }

53  std::pair<std::size_t, std::size_t> get_line_info(std::size_t offset) const {
54  auto iter = std::lower_bound(line_offsets_.begin(), line_offsets_.end(), offset);
55  if (iter == line_offsets_.end())
56  return std::make_pair(0, 0);
57 
58  std::size_t line_nbr = iter - line_offsets_.begin();
59  std::size_t char_offset = offset - *iter + 1u;
60 
61  return std::make_pair(line_nbr, char_offset);
62  }
63 
64  std::string get_location(std::size_t offset) const {
65  auto location = get_line_info(offset);
66  if (location.first == 0)
67  return file_name_ + ":?";
68  else
69  return file_name_ + ":" + utils::to_string(location.first);
70  }

72 private:

74  std::string file_name_;
75  std::string file_content_;

158  std::string attr_location =

160  if (const auto* node_attr = node.try_get_attribute(name)) {
161  gui::out << gui::warning << attr_location << ": attribute '" << name
162  << "' duplicated; only first value will be used." << std::endl;
163  gui::out << gui::warning << std::string(attr_location.size(), ' ')
164  << " first occurence at: '" << std::endl;
165  gui::out << gui::warning << std::string(attr_location.size(), ' ') << " "
166  << node_attr->get_location() << std::endl;
167  node_attr->mark_as_not_accessed();
168  continue;
169  }
170 
171  auto& attrib = node.add_attribute();
172  attrib.set_location(std::move(attr_location));
173  attrib.set_value_location(
174  file.get_location(elem_node.val().data() - tree.arena().data()));
175  attrib.set_name(std::move(name));
176  attrib.set_value(to_string(elem_node.val()));
177  break;
178  }
179  case ryml::KEYMAP: [[fallthrough]];
180  case ryml::MAP: [[fallthrough]];
181  case ryml::KEYSEQ: {
182  auto& child = node.add_child();
183  set_node(file, tree, child, elem_node);
184  break;
185  }
186  default: {
187  gui::out << gui::warning << location << ": unsupported YAML node type: '"

190  }
191  }

197  file_line_mappings file(file_name);
198  if (!file.is_open()) {
199  gui::out << gui::error << file_name << ": could not open file for parsing." << std::endl;
200  return;
201  }
202 
203  layout_node root;
204  bool parsed = false;
205 
206  const std::string extension = utils::get_file_extension(file_name);
207 
208 #if defined(LXGUI_ENABLE_XML_PARSER)
209  if (extension == ".xml") {
210  const unsigned int options = pugi::parse_ws_pcdata_single;
211 
212  pugi::xml_document doc;
213  pugi::xml_parse_result result =
214  doc.load_buffer(file.get_content().c_str(), file.get_content().size(), options);
215 
216  if (!result) {
217  gui::out << gui::error << file.get_location(result.offset) << ": "
218  << result.description() << std::endl;
219  return;
220  }
221 
222  set_node(file, root, doc.first_child());
223  parsed = true;
224  }
225 #endif
226 
227 #if defined(LXGUI_ENABLE_YAML_PARSER)
228  if (extension == ".yml" || extension == ".yaml") {
229  ryml::Tree tree = ryml::parse(ryml::to_csubstr(file.get_content()));
230  set_node(file, tree, root, tree.rootref().first_child());
231  parsed = true;
232  }
233 #endif
234 
235  if (!parsed) {
236  gui::out << gui::error << file_name
237  << ": no parser registered for extension '" + extension + "'." << std::endl;
238  return;
239  }
240 
241  for (const auto& node : root.get_children()) {
242  if (node.get_name() == "Script") {
243  std::string script_file =
244  add_on.directory + "/" + node.get_attribute_value<std::string>("file");
245 
246  try {
247  lua_.do_file(script_file);
248  } catch (const sol::error& e) {
249  std::string err = e.what();
250  gui::out << gui::error << err << std::endl;
251  event_emitter_.fire_event("LUA_ERROR", {err});
252  }
253  } else if (node.get_name() == "Include") {
254  parse_layout_file_(
255  add_on.directory + "/" + node.get_attribute_value<std::string>("file"), add_on);
256  } else {
257  try {
258  auto attr = parse_core_attributes(
259  root_.get_registry(), virtual_root_.get_registry(), node, nullptr);
260 
261  utils::observer_ptr<frame> obj;
262  auto parent = attr.parent; // copy here to prevent use-after-move
263  if (parent) {
264  obj = parent->create_child(std::move(attr));
265  } else {
266  if (attr.is_virtual)
267  obj = virtual_root_.create_root_frame(std::move(attr));
268  else
269  obj = root_.create_root_frame(std::move(attr));
270  }
271 
272  if (!obj)
273  continue;
274 
275  obj->set_addon(&add_on);
276  obj->parse_layout(node);
277  obj->notify_loaded();
278  } catch (const utils::exception& e) {
279  gui::out << gui::error << e.get_description() << std::endl;
280  }
281  }
282  }
283 
284  warn_for_not_accessed_node(root);
285 }
