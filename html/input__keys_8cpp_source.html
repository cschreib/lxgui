166 key get_key_from_codename(std::string_view sKey)
167 {
168  using index_t = std::underlying_type_t<key>;
169 
170  for (index_t i = 0; i < static_cast<index_t>(key::K_MAXKEY); ++i)
171  {
172  if (sKey == get_key_codename(static_cast<key>(i)))
173  return static_cast<key>(i);
174  }
175 
176  return key::K_UNASSIGNED;
177 }
