158 key get_key_from_codename(std::string_view key_name) {
159  using index_t = std::underlying_type_t<key>;
160 
161  for (index_t i = 0; i < static_cast<index_t>(key::k_maxkey); ++i) {
162  if (key_name == get_key_codename(static_cast<key>(i)))
163  return static_cast<key>(i);
164  }
165 
166  return key::k_unassigned;
167 }
