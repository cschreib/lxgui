110 {

112 
113  bounds2f mStartQuad(0, fWidth, 0, fHeight);
114  if (empty())
115  return mStartQuad;
116 
117  const float fAtlasWidth = get_width();
118  const float fAtlasHeight = get_height();
119 
120  std::vector<bounds2f> lOccupiedSpace;
121  lOccupiedSpace.reserve(lTextureList_.size());
122 
123  float fMaxWidth = 0.0f;
124  float fMaxHeight = 0.0f;
125 
126  auto apply_padding = [&](bounds2f mRect)
127  {
128  mRect.right += fPadding;
129  mRect.bottom += fPadding;
130  return mRect;
131  };
132 
133  for (const auto& pMat : lTextureList_)
134  {
135  if (std::shared_ptr<gui::material> pLock = pMat.second.lock())
136  {
137  lOccupiedSpace.push_back(apply_padding(pLock->get_rect()));
138  fMaxWidth = std::max(fMaxWidth, lOccupiedSpace.back().right);
139  fMaxHeight = std::max(fMaxHeight, lOccupiedSpace.back().bottom);
140  }
141  }
142 
143  for (const auto& pFont : lFontList_)
144  {
145  if (std::shared_ptr<gui::font> pLock = pFont.second.lock())
146  {
147  lOccupiedSpace.push_back(apply_padding(pLock->get_texture().lock()->get_rect()));
148  fMaxWidth = std::max(fMaxWidth, lOccupiedSpace.back().right);
149  fMaxHeight = std::max(fMaxHeight, lOccupiedSpace.back().bottom);
150  }
151  }
152 
153  float fBestArea = std::numeric_limits<float>::infinity();
154  bounds2f mBestQuad;
155 
156  for (const auto& mRectSource : lOccupiedSpace)
157  {
158  auto mTestPosition = [&](const vector2f& mPos)
159  {
160  const bounds2f mTestQuad = mStartQuad + mPos;
161  if (mTestQuad.right > fAtlasWidth || mTestQuad.bottom > fAtlasHeight)
162  return;
163 
164  const float fNewMaxWidth = std::max(fMaxWidth, mTestQuad.right);
165  const float fNewMaxHeight = std::max(fMaxHeight, mTestQuad.bottom);
166  const float fNewArea = fNewMaxWidth*fNewMaxHeight;
167 
168  if (fNewArea >= fBestArea)
169  return;
170 
171  for (const auto& mRectOther : lOccupiedSpace)
172  {
173  if (mTestQuad.overlaps(mRectOther))
174  return;
175  }
176 
177  fBestArea = fNewArea;
178  mBestQuad = mTestQuad;
179  };
180 
181  mTestPosition(mRectSource.top_right());
182  mTestPosition(mRectSource.bottom_left());
183  }
184 
185  if (std::isfinite(fBestArea))
186  return mBestQuad;
187  else
188  return std::nullopt;
