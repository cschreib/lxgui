104 std::optional<bounds2f> atlas_page::find_location_(float fWidth, float fHeight) const
105 {
106  constexpr float fPadding = 1.0f; // pixels
107 
108  bounds2f mStartQuad(0, fWidth, 0, fHeight);
109  if (empty())
110  return mStartQuad;
111 
112  const float fAtlasWidth = get_width();
113  const float fAtlasHeight = get_height();
114 
115  std::vector<bounds2f> lOccupiedSpace;
116  lOccupiedSpace.reserve(lTextureList_.size());
117 
118  float fMaxWidth = 0.0f;
119  float fMaxHeight = 0.0f;
120 
121  auto apply_padding = [&](bounds2f mRect)
122  {
123  mRect.right += fPadding;
124  mRect.bottom += fPadding;
125  return mRect;
126  };
127 
128  for (const auto& pMat : lTextureList_)
129  {
130  if (std::shared_ptr<gui::material> pLock = pMat.second.lock())
131  {
132  lOccupiedSpace.push_back(apply_padding(pLock->get_rect()));
133  fMaxWidth = std::max(fMaxWidth, lOccupiedSpace.back().right);
134  fMaxHeight = std::max(fMaxHeight, lOccupiedSpace.back().bottom);
135  }
136  }
137 
138  for (const auto& pFont : lFontList_)
139  {
140  if (std::shared_ptr<gui::font> pLock = pFont.second.lock())
141  {
142  lOccupiedSpace.push_back(apply_padding(pLock->get_texture().lock()->get_rect()));
143  fMaxWidth = std::max(fMaxWidth, lOccupiedSpace.back().right);
144  fMaxHeight = std::max(fMaxHeight, lOccupiedSpace.back().bottom);
145  }
146  }
147 
148  float fBestArea = std::numeric_limits<float>::infinity();
149  bounds2f mBestQuad;
150 
151  for (const auto& mRectSource : lOccupiedSpace)
152  {
153  auto mTestPosition = [&](const vector2f& mPos)
154  {
155  const bounds2f mTestQuad = mStartQuad + mPos;
156  if (mTestQuad.right > fAtlasWidth || mTestQuad.bottom > fAtlasHeight)
157  return;
158 
159  const float fNewMaxWidth = std::max(fMaxWidth, mTestQuad.right);
160  const float fNewMaxHeight = std::max(fMaxHeight, mTestQuad.bottom);
161  const float fNewArea = fNewMaxWidth*fNewMaxHeight;
162 
163  if (fNewArea >= fBestArea)
164  return;
165 
166  for (const auto& mRectOther : lOccupiedSpace)
167  {
168  if (mTestQuad.overlaps(mRectOther))
169  return;
170  }
171 
172  fBestArea = fNewArea;
173  mBestQuad = mTestQuad;
174  };
175 
176  mTestPosition(mRectSource.top_right());
177  mTestPosition(mRectSource.bottom_left());
178  }
179 
180  if (std::isfinite(fBestArea))
181  return mBestQuad;
182  else
183  return std::nullopt;
184 }
