28  std::visit([&](const auto& mData)

48  }
49  sStr << sTab << " #-###\n";

53  sStr << sTab << " # Color : " << mData << "\n";

56 
57  sStr << sTab << " # Tex. coord. :\n";

61  sStr << sTab << " | # bottom-right : (" << mQuad_.v[2].uvs << ")\n";
62  sStr << sTab << " | # bottom-left : (" << mQuad_.v[3].uvs << ")\n";
63  sStr << sTab << " #-###\n";

70  case blend_mode::BLEND : sStr << "BLEND\n"; break;
