112 frame_renderer::find_topmost_frame(const std::function<bool(const frame&)>& predicate) const {
113  // Iterate through the frames in reverse order from rendering (frame on top goes first)
114  for (const auto& strata_obj : utils::range::reverse(strata_list_)) {
115  for (const auto& level_obj : utils::range::reverse_value(strata_obj.level_list)) {
116  for (const auto& obj : utils::range::reverse(level_obj.frame_list)) {
117  if (const frame* raw_ptr = obj.get()) {
118  if (raw_ptr->is_visible()) {
119  if (auto topmost = raw_ptr->find_topmost_frame(predicate))
120  return topmost;
121  }
122  }
123  }
124  }
125  }
126 
127  return nullptr;
128 }
