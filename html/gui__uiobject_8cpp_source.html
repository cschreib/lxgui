777 void uiobject::read_anchors_(float& iLeft, float& iRight, float& iTop,
778  float& iBottom, float& iXCenter, float& iYCenter) const
779 {
780  iLeft = +std::numeric_limits<float>::infinity();
781  iRight = -std::numeric_limits<float>::infinity();
782  iTop = +std::numeric_limits<float>::infinity();
783  iBottom = -std::numeric_limits<float>::infinity();
784 
785  for (const auto& mOptAnchor : lAnchorList_)
786  {
787  if (!mOptAnchor)
788  continue;
789 
790  const anchor& mAnchor = mOptAnchor.value();
791 
792  // Make sure the anchored object has its borders updated
793  const uiobject* pObj = mAnchor.get_parent();
794  if (pObj)
795  pObj->update_borders_();
796 
797  switch (mAnchor.get_point())
798  {
799  case anchor_point::TOPLEFT :
800  iTop = std::min<float>(iTop, mAnchor.get_abs_y());
801  iLeft = std::min<float>(iLeft, mAnchor.get_abs_x());
802  break;
803  case anchor_point::TOP :
804  iTop = std::min<float>(iTop, mAnchor.get_abs_y());
805  iXCenter = mAnchor.get_abs_x();
806  break;
807  case anchor_point::TOPRIGHT :
808  iTop = std::min<float>(iTop, mAnchor.get_abs_y());
809  iRight = std::max<float>(iRight, mAnchor.get_abs_x());
810  break;
811  case anchor_point::RIGHT :
812  iRight = std::max<float>(iRight, mAnchor.get_abs_x());
813  iYCenter = mAnchor.get_abs_y();
814  break;
815  case anchor_point::BOTTOMRIGHT :
816  iBottom = std::max<float>(iBottom, mAnchor.get_abs_y());
817  iRight = std::max<float>(iRight, mAnchor.get_abs_x());
818  break;
819  case anchor_point::BOTTOM :
820  iBottom = std::max<float>(iBottom, mAnchor.get_abs_y());
821  iXCenter = mAnchor.get_abs_x();
822  break;
823  case anchor_point::BOTTOMLEFT :
824  iBottom = std::max<float>(iBottom, mAnchor.get_abs_y());
825  iLeft = std::min<float>(iLeft, mAnchor.get_abs_x());
826  break;
827  case anchor_point::LEFT :
828  iLeft = std::min<float>(iLeft, mAnchor.get_abs_x());
829  iYCenter = mAnchor.get_abs_y();
830  break;
831  case anchor_point::CENTER :
832  iXCenter = mAnchor.get_abs_x();
833  iYCenter = mAnchor.get_abs_y();
834  break;
835  }
836  }

848  bReady_ = true;

892 void uiobject::update_anchors()
893 {
894  if (!bUpdateAnchors_)
895  return;
896 
897  std::vector<const uiobject*> lAnchorParentList;
898  for (auto& mAnchor : lAnchorList_)
899  {
900  if (!mAnchor)
901  continue;
902 
903  const uiobject* pObj = mAnchor->get_parent();
904  if (pObj)
905  {
906  if (pObj->depends_on(this))
907  {
908  gui::out << gui::error << "gui::" << lType_.back() << " : Cyclic anchor dependency ! "
909  << "\"" << sName_ << "\" and \"" << pObj->get_name() << "\" depend on "
910  "eachothers (directly or indirectly).\n\""
911  << anchor::get_string_point(mAnchor->get_point()) << "\" anchor removed." << std::endl;
912 
913  mAnchor.reset();
914  continue;
915  }
916 
917  if (utils::find(lAnchorParentList, pObj) == lAnchorParentList.end())
918  lAnchorParentList.push_back(pObj);
919  }
920  }
921 
922  for (const auto* pParent : lPreviousAnchorParentList_)
923  {
924  if (utils::find(lAnchorParentList, pParent) == lAnchorParentList.end())
925  pParent->notify_anchored_object(this, false);
926  }
927 
928  for (const auto* pParent : lAnchorParentList)
929  {
930  if (utils::find(lPreviousAnchorParentList_, pParent) == lPreviousAnchorParentList_.end())
931  pParent->notify_anchored_object(this, true);
932  }
933 
934  lPreviousAnchorParentList_ = std::move(lAnchorParentList);
935  bUpdateAnchors_ = false;
936 }
