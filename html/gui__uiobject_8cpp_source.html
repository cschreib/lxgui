29 uiobject::~uiobject()
30 {
31  if (!bVirtual_)
32  {
33  // Tell this widget's anchor parents that it is no longer anchored to them
34  for (auto& mAnchor : lAnchorList_)
35  {
36  if (mAnchor)
37  {
38  if (auto* pAnchorParent = mAnchor->get_parent().get())
39  pAnchorParent->remove_anchored_object(*this);
40  }
41 
42  mAnchor.reset();
43  }
44 
45  // Replace anchors pointing to this widget by absolute anchors
46  // (need to copy the anchored object list, because the objects will attempt to
47  // modify it when un-anchored, which would invalidate our iteration)
48  std::vector<utils::observer_ptr<uiobject>> lTempAnchoredObjectList =
49  std::move(lAnchoredObjectList_);
50  for (const auto& pObj : lTempAnchoredObjectList)
51  {
52  if (!pObj)
53  continue;
54 
55  std::vector<anchor_point> lAnchoredPointList;
56  for (const auto& mAnchor : pObj->get_point_list())
57  {
58  if (mAnchor && mAnchor->get_parent().get() == this)
59  lAnchoredPointList.push_back(mAnchor->mPoint);
60  }
61 
62  for (const auto& mPoint : lAnchoredPointList)
63  {
64  const anchor& mAnchor = pObj->get_point(mPoint);
65  anchor_data mNewAnchor = anchor_data(mPoint, "", anchor_point::TOPLEFT);
66  mNewAnchor.mOffset = mAnchor.mOffset;
67 
68  switch (mAnchor.mParentPoint)
69  {
70  case anchor_point::TOPLEFT : mNewAnchor.mOffset += lBorderList_.top_left(); break;
71  case anchor_point::TOP : mNewAnchor.mOffset.y += lBorderList_.top; break;
72  case anchor_point::TOPRIGHT : mNewAnchor.mOffset += lBorderList_.top_right(); break;
73  case anchor_point::RIGHT : mNewAnchor.mOffset.x += lBorderList_.right; break;
74  case anchor_point::BOTTOMRIGHT : mNewAnchor.mOffset += lBorderList_.bottom_right(); break;
75  case anchor_point::BOTTOM : mNewAnchor.mOffset.y += lBorderList_.bottom; break;
76  case anchor_point::BOTTOMLEFT : mNewAnchor.mOffset += lBorderList_.bottom_left(); break;
77  case anchor_point::LEFT : mNewAnchor.mOffset.x += lBorderList_.left; break;
78  case anchor_point::CENTER : mNewAnchor.mOffset += lBorderList_.center(); break;
79  }
80 
81  pObj->set_point(mNewAnchor);
82  }
83 
84  pObj->update_anchors_();
85  }
86 
87  remove_glue();
88  }
89 
90  // Unregister this object from the GUI manager
91  if (!is_virtual() || pParent_ == nullptr)
92  get_registry().remove_uiobject(*this);
93 }

645 bool uiobject::make_borders_(float& fMin, float& fMax, float fCenter, float fSize) const
646 {
647  if (std::isinf(fMin) && std::isinf(fMax))
648  {
649  if (!std::isinf(fSize) && fSize > 0.0f && !std::isinf(fCenter))
650  {
651  fMin = fCenter - fSize/2.0f;
652  fMax = fCenter + fSize/2.0f;
653  }
654  else
655  return false;
656  }
657  else if (std::isinf(fMax))
658  {
659  if (!std::isinf(fSize) && fSize > 0.0f)
660  fMax = fMin + fSize;
661  else if (!std::isinf(fCenter))
662  fMax = fMin + 2.0f*(fCenter - fMin);
663  else
664  return false;
665  }
666  else if (std::isinf(fMin))
667  {
668  if (!std::isinf(fSize) && fSize > 0.0f)
669  fMin = fMax - fSize;
670  else if (!std::isinf(fCenter))
671  fMin = fMax - 2.0f*(fMax - fCenter);
672  else
673  return false;
674  }
675 
676  return true;
677 }

679 void uiobject::read_anchors_(float& fLeft, float& fRight, float& fTop,
680  float& fBottom, float& fXCenter, float& fYCenter) const
681 {
682  fLeft = +std::numeric_limits<float>::infinity();
683  fRight = -std::numeric_limits<float>::infinity();
684  fTop = +std::numeric_limits<float>::infinity();
685  fBottom = -std::numeric_limits<float>::infinity();
686 
687  for (const auto& mOptAnchor : lAnchorList_)
688  {
689  if (!mOptAnchor)
690  continue;
691 
692  const anchor& mAnchor = mOptAnchor.value();
693  const vector2f mAnchorPoint = mAnchor.get_point(*this);
694 
695  switch (mAnchor.mPoint)
696  {
697  case anchor_point::TOPLEFT :
698  fTop = std::min<float>(fTop, mAnchorPoint.y);
699  fLeft = std::min<float>(fLeft, mAnchorPoint.x);
700  break;
701  case anchor_point::TOP :
702  fTop = std::min<float>(fTop, mAnchorPoint.y);
703  fXCenter = mAnchorPoint.x;
704  break;
705  case anchor_point::TOPRIGHT :
706  fTop = std::min<float>(fTop, mAnchorPoint.y);
707  fRight = std::max<float>(fRight, mAnchorPoint.x);
708  break;
709  case anchor_point::RIGHT :
710  fRight = std::max<float>(fRight, mAnchorPoint.x);
711  fYCenter = mAnchorPoint.y;
712  break;
713  case anchor_point::BOTTOMRIGHT :
714  fBottom = std::max<float>(fBottom, mAnchorPoint.y);
715  fRight = std::max<float>(fRight, mAnchorPoint.x);
716  break;
717  case anchor_point::BOTTOM :
718  fBottom = std::max<float>(fBottom, mAnchorPoint.y);
719  fXCenter = mAnchorPoint.x;
720  break;
721  case anchor_point::BOTTOMLEFT :
722  fBottom = std::max<float>(fBottom, mAnchorPoint.y);
723  fLeft = std::min<float>(fLeft, mAnchorPoint.x);
724  break;
725  case anchor_point::LEFT :
726  fLeft = std::min<float>(fLeft, mAnchorPoint.x);
727  fYCenter = mAnchorPoint.y;
728  break;
729  case anchor_point::CENTER :
730  fXCenter = mAnchorPoint.x;
731  fYCenter = mAnchorPoint.y;
732  break;
733  }
734  }

811 void uiobject::update_anchors_()
812 {
813  std::vector<utils::observer_ptr<uiobject>> lAnchorParentList;
814  for (auto& mAnchor : lAnchorList_)
815  {
816  if (!mAnchor)
817  continue;
818 
819  utils::observer_ptr<uiobject> pObj = mAnchor->get_parent();
820  if (pObj)
821  {
822  if (pObj->depends_on(*this))
823  {
824  gui::out << gui::error << "gui::" << lType_.back() << " : Cyclic anchor dependency ! "
825  << "\"" << sName_ << "\" and \"" << pObj->get_name() << "\" depend on "
826  "eachothers (directly or indirectly).\n\""
827  << anchor::get_string_point(mAnchor->mPoint) << "\" anchor removed." << std::endl;
828 
829  mAnchor.reset();
830  continue;
831  }
832 
833  if (utils::find(lAnchorParentList, pObj) == lAnchorParentList.end())
834  lAnchorParentList.push_back(pObj);
835  }
836  }
837 
838  for (const auto& pParent : lPreviousAnchorParentList_)
839  {
840  if (utils::find(lAnchorParentList, pParent) == lAnchorParentList.end())
841  pParent->remove_anchored_object(*this);
842  }
843 
844  for (const auto& pParent : lAnchorParentList)
845  {
846  if (utils::find(lPreviousAnchorParentList_, pParent) == lPreviousAnchorParentList_.end())
847  pParent->add_anchored_object(*this);
848  }
849 
850  lPreviousAnchorParentList_ = std::move(lAnchorParentList);
851 }
