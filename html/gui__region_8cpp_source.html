30 region::~region()
31 {
32  if (!bVirtual_)
33  {
34  // Tell this region's anchor parents that it is no longer anchored to them
35  for (auto& mAnchor : lAnchorList_)
36  {
37  if (mAnchor)
38  {
39  if (auto* pAnchorParent = mAnchor->get_parent().get())
40  pAnchorParent->remove_anchored_object(*this);
41  }
42 
43  mAnchor.reset();
44  }
45 
46  // Replace anchors pointing to this region by absolute anchors
47  // (need to copy the anchored object list, because the objects will attempt to
48  // modify it when un-anchored, which would invalidate our iteration)
49  std::vector<utils::observer_ptr<region>> lTempAnchoredObjectList =
50  std::move(lAnchoredObjectList_);
51  for (const auto& pObj : lTempAnchoredObjectList)
52  {
53  if (!pObj)
54  continue;
55 
56  std::vector<anchor_point> lAnchoredPointList;
57  for (const auto& mAnchor : pObj->get_point_list())
58  {
59  if (mAnchor && mAnchor->get_parent().get() == this)
60  lAnchoredPointList.push_back(mAnchor->mPoint);
61  }
62 
63  for (const auto& mPoint : lAnchoredPointList)
64  {
65  const anchor& mAnchor = pObj->get_point(mPoint);
66  anchor_data mNewAnchor = anchor_data(mPoint, "", anchor_point::TOP_LEFT);
67  mNewAnchor.mOffset = mAnchor.mOffset;
68 
69  switch (mAnchor.mParentPoint)
70  {
71  case anchor_point::TOP_LEFT : mNewAnchor.mOffset += lBorderList_.top_left(); break;
72  case anchor_point::TOP : mNewAnchor.mOffset.y += lBorderList_.top; break;
73  case anchor_point::TOP_RIGHT : mNewAnchor.mOffset += lBorderList_.top_right(); break;
74  case anchor_point::RIGHT : mNewAnchor.mOffset.x += lBorderList_.right; break;
75  case anchor_point::BOTTOM_RIGHT : mNewAnchor.mOffset += lBorderList_.bottom_right(); break;
76  case anchor_point::BOTTOM : mNewAnchor.mOffset.y += lBorderList_.bottom; break;
77  case anchor_point::BOTTOM_LEFT : mNewAnchor.mOffset += lBorderList_.bottom_left(); break;
78  case anchor_point::LEFT : mNewAnchor.mOffset.x += lBorderList_.left; break;
79  case anchor_point::CENTER : mNewAnchor.mOffset += lBorderList_.center(); break;
80  }
81 
82  pObj->set_point(mNewAnchor);
83  }
84 
85  pObj->update_anchors_();
86  }
87 
88  remove_glue();
89  }
90 
91  // Unregister this object from the GUI manager
92  if (!is_virtual() || pParent_ == nullptr)
93  get_registry().remove_region(*this);
94 }

652 bool region::make_borders_(float& fMin, float& fMax, float fCenter, float fSize) const
653 {
654  if (std::isinf(fMin) && std::isinf(fMax))
655  {
656  if (!std::isinf(fSize) && fSize > 0.0f && !std::isinf(fCenter))
657  {
658  fMin = fCenter - fSize/2.0f;
659  fMax = fCenter + fSize/2.0f;
660  }
661  else
662  return false;
663  }
664  else if (std::isinf(fMax))
665  {
666  if (!std::isinf(fSize) && fSize > 0.0f)
667  fMax = fMin + fSize;
668  else if (!std::isinf(fCenter))
669  fMax = fMin + 2.0f*(fCenter - fMin);
670  else
671  return false;
672  }
673  else if (std::isinf(fMin))
674  {
675  if (!std::isinf(fSize) && fSize > 0.0f)
676  fMin = fMax - fSize;
677  else if (!std::isinf(fCenter))
678  fMin = fMax - 2.0f*(fMax - fCenter);
679  else
680  return false;
681  }
682 
683  return true;
684 }

686 void region::read_anchors_(float& fLeft, float& fRight, float& fTop,
687  float& fBottom, float& fXCenter, float& fYCenter) const
688 {
689  fLeft = +std::numeric_limits<float>::infinity();
690  fRight = -std::numeric_limits<float>::infinity();
691  fTop = +std::numeric_limits<float>::infinity();
692  fBottom = -std::numeric_limits<float>::infinity();
693 
694  for (const auto& mOptAnchor : lAnchorList_)
695  {
696  if (!mOptAnchor)
697  continue;
698 
699  const anchor& mAnchor = mOptAnchor.value();
700  const vector2f mAnchorPoint = mAnchor.get_point(*this);
701 
702  switch (mAnchor.mPoint)
703  {
704  case anchor_point::TOP_LEFT :
705  fTop = std::min<float>(fTop, mAnchorPoint.y);
706  fLeft = std::min<float>(fLeft, mAnchorPoint.x);
707  break;
708  case anchor_point::TOP :
709  fTop = std::min<float>(fTop, mAnchorPoint.y);
710  fXCenter = mAnchorPoint.x;
711  break;
712  case anchor_point::TOP_RIGHT :
713  fTop = std::min<float>(fTop, mAnchorPoint.y);
714  fRight = std::max<float>(fRight, mAnchorPoint.x);
715  break;
716  case anchor_point::RIGHT :
717  fRight = std::max<float>(fRight, mAnchorPoint.x);
718  fYCenter = mAnchorPoint.y;
719  break;
720  case anchor_point::BOTTOM_RIGHT :
721  fBottom = std::max<float>(fBottom, mAnchorPoint.y);
722  fRight = std::max<float>(fRight, mAnchorPoint.x);
723  break;
724  case anchor_point::BOTTOM :
725  fBottom = std::max<float>(fBottom, mAnchorPoint.y);
726  fXCenter = mAnchorPoint.x;
727  break;
728  case anchor_point::BOTTOM_LEFT :
729  fBottom = std::max<float>(fBottom, mAnchorPoint.y);
730  fLeft = std::min<float>(fLeft, mAnchorPoint.x);
731  break;
732  case anchor_point::LEFT :
733  fLeft = std::min<float>(fLeft, mAnchorPoint.x);
734  fYCenter = mAnchorPoint.y;
735  break;
736  case anchor_point::CENTER :
737  fXCenter = mAnchorPoint.x;
738  fYCenter = mAnchorPoint.y;
739  break;
740  }
741  }

818 void region::update_anchors_()
819 {
820  std::vector<utils::observer_ptr<region>> lAnchorParentList;
821  for (auto& mAnchor : lAnchorList_)
822  {
823  if (!mAnchor)
824  continue;
825 
826  utils::observer_ptr<region> pObj = mAnchor->get_parent();
827  if (pObj)
828  {
829  if (pObj->depends_on(*this))
830  {
831  gui::out << gui::error << "gui::" << lType_.back() << " : Cyclic anchor dependency ! "
832  << "\"" << sName_ << "\" and \"" << pObj->get_name() << "\" depend on "
833  "eachothers (directly or indirectly).\n\""
834  << anchor::get_string_point(mAnchor->mPoint) << "\" anchor removed." << std::endl;
835 
836  mAnchor.reset();
837  continue;
838  }
839 
840  if (utils::find(lAnchorParentList, pObj) == lAnchorParentList.end())
841  lAnchorParentList.push_back(pObj);
842  }
843  }
844 
845  for (const auto& pParent : lPreviousAnchorParentList_)
846  {
847  if (utils::find(lAnchorParentList, pParent) == lAnchorParentList.end())
848  pParent->remove_anchored_object(*this);
849  }
850 
851  for (const auto& pParent : lAnchorParentList)
852  {
853  if (utils::find(lPreviousAnchorParentList_, pParent) == lPreviousAnchorParentList_.end())
854  pParent->add_anchored_object(*this);
855  }
856 
857  lPreviousAnchorParentList_ = std::move(lAnchorParentList);
858 }
