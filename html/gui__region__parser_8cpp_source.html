102  std::string parent = anchor_node.get_attribute_value_or<std::string>(
103  "relativeTo", parent_ || is_virtual() ? "$parent" : "");
104  point relative_point = anchor_node.get_attribute_value_or<point>("relativePoint", pt);
105
