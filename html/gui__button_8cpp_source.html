30 void button::fire_script(const std::string& script_name, const event_data& data) {
31  alive_checker checker(*this);
32  base::fire_script(script_name, data);
33  if (!checker.is_alive())
34  return;
35 
36  if (is_enabled()) {
37  if (script_name == "OnEnter")
38  highlight();
39 
40  if (script_name == "OnLeave") {
41  unlight();
42 
43  if (state_ == state::down)
44  release();
45  }
46 
47  if (script_name == "OnMouseDown") {
48  push();
49  fire_script("OnClick");
50  if (!checker.is_alive())
51  return;
52  }
53 
54  if (script_name == "OnMouseUp")
55  release();
56  }
57 }
