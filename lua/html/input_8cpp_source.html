391 void manager::update(float fTempDelta)
392 {
393  if (bRemoveFocus_)
394  {
395  bFocus_ = false;
396  pFocusReceiver_ = nullptr;
397  bRemoveFocus_ = false;
398  }
399 
400  if (!pSource_->is_manually_updated())
401  pSource_->update();
402 
403  // Control extreme delta time after loading/at startup etc
404  double dDelta = fTempDelta;
405  if ((dDelta < 0.0) || (dDelta > 1.0))
406  dDelta = 0.05;
407 
408  // Update keys
409  const source_impl::key_state& mKeyState = pSource_->get_key_state();
410  bKey_ = false;
411  for (uint i = 0; i < KEY_NUMBER; ++i)
412  {
413  // Update delays
414  if (mKeyState.lKeyState[i])
415  {
416  bKey_ = true;
417  lKeyDelay_[i] += dDelta;
418  if (lKeyDelay_[i] >= dLongPressDelay_)
419  lKeyLong_[i] = true;
420  }
421  else
422  {
423  lKeyDelay_[i] = 0.0;
424  lKeyLong_[i] = false;
425  }
426  }
427 
428  // Handle modifier keys
429  bCtrlPressed_ = key_is_down(key::K_LCONTROL, true) || key_is_down(key::K_RCONTROL, true);
430  bShiftPressed_ = key_is_down(key::K_LSHIFT, true) || key_is_down(key::K_RSHIFT, true);
431  bAltPressed_ = key_is_down(key::K_LMENU, true) || key_is_down(key::K_RMENU, true);
432 
433  // Update mouse state
434  const source_impl::mouse_state& mMouseState = pSource_->get_mouse_state();
435  for (uint i = 0; i < MOUSE_BUTTON_NUMBER; ++i)
436  {
437  // Update delays
438  if (mMouseState.lButtonState[i])
439  {
440  lMouseDelay_[i] += dDelta;
441  if (lMouseDelay_[i] >= dLongPressDelay_)
442  lMouseLong_[i] = true;
443  }
444  else
445  {
446  lMouseDelay_[i] = 0.0;
447  lMouseLong_[i] = false;
448  }
449  }
450 
451  // Send events
452  for (auto& mEvent : pSource_->poll_events())
453  {
454  if (mEvent.get_name() == "KEY_PRESSED" || mEvent.get_name() == "KEY_RELEASED")
455  {
456  // Add key name to the event
457  mEvent.add(get_key_name(utils::get<key>(mEvent.get(0))));
458  }
459  else if (mEvent.get_name() == "MOUSE_PRESSED" ||
460  mEvent.get_name() == "MOUSE_RELEASED" ||
461  mEvent.get_name() == "MOUSE_DOUBLE_CLICKED")
462  {
463  // Add button name to the event
464  mEvent.add(get_mouse_button_string(mEvent.get<mouse_button>(0)));
465  }
466 
467  fire_event_(mEvent);
468  }
469 
470  lChars_ = pSource_->get_chars();
471  if (!lChars_.empty())
472  {
473  gui::event mCharEvent("TEXT_ENTERED");
474  mCharEvent.add(std::uint32_t{});
475  for (auto cChar : lChars_)
476  {
477  mCharEvent.get(0) = cChar;
478  fire_event_(mCharEvent);
479  }
480  }
481 
482  // Update mouse position
483  if (mMouseState.bHasDelta)
484  {
485  fDMX_ = mMouseState.fDX;
486  fDMY_ = mMouseState.fDY;
487  fRelDMX_ = mMouseState.fRelDX;
488  fRelDMY_ = mMouseState.fRelDY;
489  }
490  else
491  {
492  fDMX_ = mMouseState.fAbsX - fMX_;
493  fDMY_ = mMouseState.fAbsY - fMY_;
494  fRelDMX_ = mMouseState.fRelX - fRelMX_;
495  fRelDMY_ = mMouseState.fRelY - fRelMY_;
496  }
497 
498  fMX_ = mMouseState.fAbsX;
499  fMY_ = mMouseState.fAbsY;
500  fRelMX_ = mMouseState.fRelX;
501  fRelMY_ = mMouseState.fRelY;
502 
503  fMWheel_ = mMouseState.fRelWheel;
504  if (fMWheel_ == 0.0f)
505  bWheelRolled_ = false;
506  else
507  bWheelRolled_ = true;
508 
509  // Send movement event
510  if (fDMX_ != 0.0 || fDMY_ != 0.0)
511  {
512  gui::event mMouseMovedEvent("MOUSE_MOVED", true);
513  mMouseMovedEvent.add(fDMX_);
514  mMouseMovedEvent.add(fDMY_);
515  mMouseMovedEvent.add(fDMX_*mMouseState.fRelX/mMouseState.fAbsX);
516  mMouseMovedEvent.add(fDMY_*mMouseState.fRelY/mMouseState.fAbsY);
517  fire_event_(mMouseMovedEvent, true);
518  }
519 
520  if (bWheelRolled_)
521  {
522  gui::event mMouseWheelEvent("MOUSE_WHEEL", true);
523  mMouseWheelEvent.add(fMWheel_);
524  fire_event_(mMouseWheelEvent, true);
525  }
526 
527  if (pSource_->has_window_resized())
528  {
529  gui::event mWindowResizedEvent("WINDOW_RESIZED", true);
530  mWindowResizedEvent.add(pSource_->get_window_new_width());
531  mWindowResizedEvent.add(pSource_->get_window_new_height());
532  fire_event_(mWindowResizedEvent, true);
533  pSource_->reset_window_resized();
534  }
535 
536  dTime_ += dDelta;
537 }
